17  package org.gwe.rse.provider.xnat;
18  
19  import java.util.Map;
20  
21  import org.gwe.rse.FieldMeta;
22  import org.gwe.rse.utils.IdentifiableMap;
23  import org.gwe.rse.utils.TreeDS;
24  
25  /**
26   * @author Marco Ruiz
27   * @since Dec 19, 2009
28   */
29  public class XNATMetadata {
30  	
31  	private static String URL_PREFIX = "${server}/REST/projects";
32  	
33  /*
34  	${XNAT_SERVER} = http://central.xnat.org
35  	${PROJECT}     = CENTRAL_OASIS_CS
36  	${SUBJECT}     = OAS1_0014
37  	${EXPERIMENT}  = OAS1_0014_MR1
38  	${SCAN}        = mpr-1
39  
40  	${XNAT_SERVER}/REST/projects?format=xml
41  	${XNAT_SERVER}/REST/projects/${PROJECT}/subjects?format=xml
42  	${XNAT_SERVER}/REST/projects/${PROJECT}/subjects/${SUBJECT}/experiments?format=xml
43  	${XNAT_SERVER}/REST/projects/${PROJECT}/subjects/${SUBJECT}/experiments/${EXPERIMENT}/scans?format=xml
44  	${XNAT_SERVER}/REST/projects/${PROJECT}/subjects/${SUBJECT}/experiments/${EXPERIMENT}/scans/${SCAN}/files?format=xml
45  
46  	http://central.xnat.org/REST/projects/CENTRAL_OASIS_CS/subjects/OAS1_0014/experiments/OAS1_0014_MR1/scans/mpr-1/files?format=xml
47   */
48  
49  	private TreeDS<FieldMeta> root; 
50  	private Map<String, TreeDS<FieldMeta>> nodesMap; 
51  
52  	public XNATMetadata() {
53  		this.root = createNode("project", "Project", "?format=xml", "//row/cell[1]/text()");
54  		TreeDS<FieldMeta> currNode = root;
55  		currNode = currNode.addChild(createNode("subject", "Subject",       "/${project}/subjects?format=xml", 
56  				"//row/cell[2]/text()"));
57  		
58  		currNode = currNode.addChild(createNode("experiment", "Experiment", "/${project}/subjects/${subject}/experiments?format=xml", 
59  				"//row/cell[2]/text()"));
60  		
61  		currNode = currNode.addChild(createNode("scan", "Scan",             "/${project}/subjects/${subject}/experiments/${experiment}/scans?format=xml", 
62  				"//row/cell[1]/text()"));
63  		
64  		currNode = currNode.addChild(createNode("file", "File",             "/${project}/subjects/${subject}/experiments/${experiment}/scans/${scan}/files?format=xml", 
65  				"//row/cell[3]/text()"));
66  		
67  		this.nodesMap = IdentifiableMap.createNodesMap(this.root);
68      }
69  
70  	private TreeDS<FieldMeta> createNode(String elementId, String caption, String sufixURLTemplate, String xpathExpr) {
71  		return new TreeDS<FieldMeta>(new XNATFieldMeta(elementId, caption, "", URL_PREFIX + sufixURLTemplate, xpathExpr));
72      }
73  
74  	public TreeDS<FieldMeta> getRoot() {
75      	return root;
76      }
77  
78  	public void setRoot(TreeDS<FieldMeta> root) {
79      	this.root = root;
80      }
81  
82  	public XNATFieldMeta getFieldMeta(String fieldId) {
83  		return (XNATFieldMeta) this.nodesMap.get(fieldId).getElement();
84      }
85  }
