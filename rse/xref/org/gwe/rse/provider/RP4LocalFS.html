17  package org.gwe.rse.provider;
18  
19  import java.io.File;
20  import java.util.ArrayList;
21  import java.util.HashSet;
22  import java.util.List;
23  import java.util.Set;
24  
25  import org.gwe.rse.FieldData;
26  import org.gwe.rse.FieldDataMap;
27  import org.gwe.rse.FieldMeta;
28  import org.gwe.rse.utils.TreeDS;

30  /**
31   * @author Marco Ruiz
32   * @since Dec 11, 2009
33   */
34  public class RP4LocalFS extends TreeBasedRecordProvider {
35  	
36  	private String rootPath = "/Users";
37  	
38      public RP4LocalFS(String name) {
39      	super(name);
40      }
41  
42  	public TreeDS<FieldMeta> readMetadataTree() {
43  	    TreeDS<FieldMeta> result = new TreeDS<FieldMeta>(new FieldMeta("root", "Root", true, ""));
44  	    TreeDS<FieldMeta> currNode = result; 
45  	    
46  	    currNode = currNode.addChild(new FieldMeta("level1A", "Level 1-A", true, ""));
47  	    currNode = currNode.addChild(new FieldMeta("level2A", "Level 2-A", true, ""));
48  	    currNode = currNode.addChild(new FieldMeta("level3A", "Level 3-A", true, ""));
49  	    
50  	    currNode = result;
51  	    currNode = currNode.addChild(new FieldMeta("level1B", "Level 1-B", true, ""));
52  	    currNode = currNode.addChild(new FieldMeta("level2B", "Level 2-B", true, ""));
53  	    
54  		return result;
55      }
56  
57      public Set<FieldData> readStartUpFields() {
58      	List<FieldData> ancestors = new ArrayList<FieldData>();
59      	
60  	    Set<FieldData> result = new HashSet<FieldData>();
61  	    FieldData rootData = addDirData("root",    ancestors, "admin", result);
62  	    addDirData("level1A", ancestors, "work", result);
63  	    addDirData("level2A", ancestors, "eclipse-ws", result);
64  	    addDirData("level3A", ancestors, "gwe-core", result);
65  	    
66  	    ancestors.clear();
67  	    ancestors.add(rootData);
68  	    
69  	    addDirData("level1B", ancestors, "gwe-daemon", result);
70  	    addDirData("level2B", ancestors, "gwe-0.7.3.alpha", result);
71  		return result;
72      }
73  
74  	private FieldData addDirData(String id, List<FieldData> ancestors, String path, Set<FieldData> result) {
75  	    FieldData fldData = createDirData(id, ancestors, path);
76  		result.add(fldData);
77  	    ancestors.add(fldData);
78  	    return fldData;
79      }
80  
81      public FieldData getFieldUpdate(TreeDS<FieldData> changedNode, FieldDataMap filters, String fieldId) {
82  		return createDirData(fieldId, changedNode.getAncestorsAsList(true), "");
83      }
84      
85      private FieldData createDirData(String id, List<FieldData> ancestors, String path) {
86      	String fullPath = rootPath;
87      	for (FieldData fldData : ancestors) 
88      		fullPath += "/" + fldData.getValue();
89  
90      	List<String> subDirs = new ArrayList<String>();
91  		for (File file : new File(fullPath).listFiles()) 
92  	        if (file.isDirectory()) 
93  	        	subDirs.add(file.getName());
94  
95          return new FieldData(id, path, subDirs.toArray(new String[subDirs.size()]));
96      }
97  }
