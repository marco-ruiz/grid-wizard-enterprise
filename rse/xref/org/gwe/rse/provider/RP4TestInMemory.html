32  public class RP4TestInMemory extends AbstractRecordProvider {

34  	private int counter = 0;

36  	public RP4TestInMemory(String name) {
37  		super(name);
38      }
39  
40  	public List<? extends FieldMeta> readMetadata() {
41  		List<FieldMeta> meta = new ArrayList<FieldMeta>();
42  		meta.add(new FieldMeta("id"   , "ID"     , false, "text"));
43  		meta.add(new FieldMeta("name" , "NAME"   , false, "text"));
44  		meta.add(new FieldMeta("i"    , "index 1", true , "text"));
45  		meta.add(new FieldMeta("image", "Image"  , false, "img"));
46  		meta.add(new FieldMeta("j"    , "Index 2", true , "text"));
47  		meta.add(new FieldMeta("k"    , "Index 3", true , "text"));
48  		meta.add(new FieldMeta("cmd"  , "Command", false, "text"));
49  		meta.add(new FieldMeta("l"    , "Index 4", true , "text"));
50  		return meta;
51      }
52  	
53  	public Set<FieldData> readStartUpFields() {
54  		return createRecord("echo 'test'", 99);
55      }
56  
57      public Set<FieldData> readRecord(String fieldId, int valueOptionIndex, boolean filterAllValues) {
58  		return createRecord(fieldId, valueOptionIndex);
59      }
60  	
61  	private Set<FieldData> createRecord(Object cmdValue, int limit) {
62  		Set<FieldData> data = new HashSet<FieldData>();
63  		data.add(new FieldData("id"   , Math.random() + ""));
64  		data.add(new FieldData("name" , "Marco"   ));
65  		data.add(new FieldData("i"    , "11"         , range(0, limit * 50, 2)));
66  		data.add(new FieldData("image", counter++ + ".jpg"  ));
67  		data.add(new FieldData("j"    , "22"         , range(0, 88, 11)));
68  		data.add(new FieldData("k"    , "33"         , range(20, 50, 1)));
69  		data.add(new FieldData("cmd"  , cmdValue.toString()));
70  		data.add(new FieldData("l"    , "44"         , range(4, 100, 4)));
71  	    return data;
72      }
73  
74  	public List<String> range(int start, int end, int step) {
75  		List<String> result = new ArrayList<String>();
76  		for (int idx = start; idx < end; idx += step) result.add(idx + "");
77  		return result;
78  	}
