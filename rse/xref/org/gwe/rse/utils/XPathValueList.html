17  package org.gwe.rse.utils;
18  
19  import java.io.InputStream;
20  import java.io.StringWriter;
21  import java.util.ArrayList;
22  
23  import javax.xml.transform.OutputKeys;
24  import javax.xml.transform.Transformer;
25  import javax.xml.transform.TransformerException;
26  import javax.xml.transform.TransformerFactory;
27  import javax.xml.transform.TransformerFactoryConfigurationError;
28  import javax.xml.transform.dom.DOMSource;
29  import javax.xml.transform.stream.StreamResult;
30  import javax.xml.xpath.XPath;
31  import javax.xml.xpath.XPathConstants;
32  import javax.xml.xpath.XPathFactory;
33  
34  import org.w3c.dom.Node;
35  import org.w3c.dom.NodeList;
36  import org.xml.sax.InputSource;
37  
38  /**
39   * @author Marco Ruiz
40   * @since Dec 14, 2009
41   */
42  public class XPathValueList extends ArrayList<String> {
43  
44  	public XPathValueList(InputStream source, String expression) {
45          XPath xpath = XPathFactory.newInstance().newXPath();
46  		try {
47  			InputSource inputSource = new InputSource(source);
48  	        NodeList queryResult = (NodeList) xpath.evaluate(expression, inputSource, XPathConstants.NODESET);
49  	        int length = queryResult.getLength();
50  			for (int idx = 0; idx < length; idx++) {
51  		        Node item = queryResult.item(idx);
52  				String value = (item.getNodeType() == Node.ATTRIBUTE_NODE || expression.endsWith("text()")) ? 
53  						item.getTextContent() : createXMLString(item);
54  				value = value.replaceAll("^\\s*", "").replaceAll("\\s*$", "");
55  				add(value);
56              }
57  			if (size() == 0) add("");
58          } catch (Exception e) {
59          	e.printStackTrace();
60          }
61      }
62  
63  	private String createXMLString(Node source) throws TransformerFactoryConfigurationError, TransformerException {
64      	Transformer transformer = TransformerFactory.newInstance().newTransformer();
65      	transformer.setOutputProperty(OutputKeys.INDENT, "yes");
66  
67      	StreamResult result = new StreamResult(new StringWriter());
68      	transformer.transform(new DOMSource(source), result);
69  
70      	return result.getWriter().toString();
71      }
72  }
