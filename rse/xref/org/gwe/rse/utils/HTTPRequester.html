17  package org.gwe.rse.utils;
18  
19  import java.io.IOException;
20  import java.io.Reader;
21  import java.io.StringWriter;
22  import java.io.Writer;
23  import java.net.HttpURLConnection;
24  import java.net.MalformedURLException;
25  import java.net.ProtocolException;
26  import java.net.URL;
27  
28  /**
29   * @author Marco Ruiz
30   * @since Dec 20, 2009
31   */
32  public class HTTPRequester {
33  	
34  	private URL targetURL;
35  	private String auth = null;
36  	private String requestMethod;
37  	
38  	public HTTPRequester(String targetURL) throws MalformedURLException {
39  		this(targetURL, true);
40      }
41  
42  	public HTTPRequester(String targetURL, boolean postMethod) throws MalformedURLException {
43  		this.targetURL = new URL(targetURL);
44      	this.requestMethod = postMethod ? "POST" : "GET";
45      }
46  
47  	public void setAuth(String user, String password) throws Exception {
48  	    this.auth = IOStaticUtils.buildURLAuth(user, password);
49  	}
50  
51  	public Writer connect() throws Exception {
52  		return connect(new StringWriter(), null);
53  	}
54  
55  	public Writer connect(Writer output) throws Exception {
56  		return connect(output, null);
57  	}
58  	
59  	public Writer connect(Writer output, Reader input) throws Exception {
60  		HttpURLConnection urlc = null;
61  		try {
62  			urlc = createPostConnection();
63              if (input  != null) IOStaticUtils.writeToConnection(urlc, input);
64              if (output != null) IOStaticUtils.readFromConnection(urlc, output);
65  		} catch (IOException e) {
66  			throw new Exception("Cannot connect to " + targetURL, e);
67  		} finally {
68  			if (urlc != null) urlc.disconnect();
69  		}
70  		return output;
71  	}
72  
73  	private HttpURLConnection createPostConnection() throws IOException, Exception {
74  	    HttpURLConnection conn = (HttpURLConnection) targetURL.openConnection();
75  	    if (auth != null && !auth.equals(""))
76  	    	conn.setRequestProperty("Authorization", "Basic " + auth);
77  
78  	    try {
79  	    	conn.setRequestMethod(requestMethod);
80  	    } catch (ProtocolException e) {
81  	    	throw new Exception(requestMethod + " not supported", e);
82  	    }
83  	    conn.setDoOutput(true);
84  	    conn.setDoInput(true);
85  	    conn.setUseCaches(false);
86  	    conn.setAllowUserInteraction(false);
87  	    conn.setRequestProperty("Content-type", "text/xml; charset=" + "UTF-8");
88  	    return conn;
89      }
90  }
