17  package org.gwe.rse.servlet.jetty;
18  
19  import java.util.ArrayList;
20  import java.util.Arrays;
21  import java.util.List;
22  
23  import org.apache.commons.logging.Log;
24  import org.apache.commons.logging.LogFactory;
25  import org.mortbay.jetty.Server;
26  import org.mortbay.jetty.handler.ContextHandlerCollection;
27  import org.mortbay.jetty.handler.DefaultHandler;
28  import org.mortbay.jetty.servlet.Context;
29  
30  /**
31   * @author Marco Ruiz
32   * @since Jan 25, 2008
33   */
34  public class WebServerApp {
35  
36  	private static Log log = LogFactory.getLog(WebServerApp.class);
37  
38  	public static void main(String[] args) throws Exception {
39  		List<String> argsList = getAsList(args);
40  		String confArg = argsList.remove(0);
41  		int port = (argsList.size() >= 1) ? Integer.parseInt(argsList.get(0)) : 8088;
42  //		String confArg = (argsList.size() >= 2) ? argsList.get(1) : "";
43  		WebServerApp app = new WebServerApp(port);
44  		app.startWebApp(confArg);
45      }
46  
47  	private static List<String> getAsList(String[] args) {
48  	    return new ArrayList<String>(Arrays.asList(args));
49      }
50  	
51      private int port;
52  	private Server server;
53  	private Context context;
54  
55      public WebServerApp(int port) {
56      	this.port = port;
57      }
58  
59  	private void startWebApp(String confArg) throws Exception {
60  		header();
61  
62  		ContextHandlerCollection contexts = new ContextHandlerCollection();
63  		context = new RSEServletContext(confArg, contexts);
64  		
65  		System.out.println("Launching...");
66  	    server = new Server(port);
67  		server.addHandler(contexts);
68  		server.addHandler(new DefaultHandler());
69  		
70  		server.setHandler(context);
71  		server.start();
72  
73  		footer();
74  //		openURL(getURL());
75  		server.join();
76  	}
77  
78  	private void header() {
79  	    System.out.println("\n==============================================================\n");
80  		System.out.println("\tWelcome to the RSE Web Application");
81  		System.out.println("\tURL: " + getURL() + " \n");
82  		System.out.println("\n==============================================================\n");
83      }
84  	
85  	private void footer() {
86  	    System.out.println("RSE Application Launched!");
87  		System.out.println("Point your browser to " + getURL() + " to access it...");
88      }
89  
90  	private String getURL() {
91  	    return "'http://localhost:" + port + "'";
92      }
93  }
