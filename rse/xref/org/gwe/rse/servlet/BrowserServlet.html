17  package org.gwe.rse.servlet;
18  
19  import java.io.IOException;
20  import java.util.Map;
21  
22  import javax.servlet.ServletContext;
23  import javax.servlet.ServletException;
24  
25  import org.gwe.rse.AbstractRecordProvider;
26  import org.gwe.rse.provider.RecordProviderFactory;
27  import org.gwe.rse.render.BrowserTemplate;
28  import org.gwe.rse.servlet.sessions.AuthRecordSetProviderRepo;
29  
30  
31  /**
32   * @author Marco Ruiz
33   * @since Dec 13, 2008
34   */
35  public abstract class BrowserServlet extends BaseRSEServlet {
36  
37      private static final String ATTR_BROWSER_TEMPLATE = "RSE_BROWSER_TEMPLATE";
38  
39  	public void init() throws ServletException {
40  	    super.init();
41  	    
42  	    ServletContext ctx = getServletContext();
43  		synchronized (ctx) {
44  			if (getVelocityTemplate() == null)
45  	            try {
46  	                ctx.setAttribute(ATTR_BROWSER_TEMPLATE, new BrowserTemplate());
47                  } catch (IOException e) {
48                  	throw new ServletException(e);
49                  }
50          }
51      }
52  
53  	protected <PROV_T extends AbstractRecordProvider> 
54  		String buildOutput(Class<PROV_T> providerClass, Map<String, String> params, AuthRecordSetProviderRepo repo) throws ServletException {
55  		repo.addProvider((PROV_T) createProvider(getProviderFactory(), providerClass, params));
56  		return buildOutput(repo.getSize() - 1, repo);
57      }
58  
59  	public synchronized <PROV_T extends AbstractRecordProvider> PROV_T createProvider(RecordProviderFactory factory, Class<PROV_T> providerClass, Map<String, String> params) {
60  	    return factory.create(providerClass, params);
61      }
62  
63  	protected String buildOutput(int srcNum, AuthRecordSetProviderRepo repo) throws ServletException {
64          try {
65  			return getVelocityTemplate().evaluate(srcNum, repo);
66          } catch (Exception e) {
67          	throw new ServletException("Problems parsing velocity template", e.getCause());
68          }
69      }
70  
71  	public BrowserTemplate getVelocityTemplate() {
72  		return getCtxAttribute(ATTR_BROWSER_TEMPLATE);
73      }
74  }
