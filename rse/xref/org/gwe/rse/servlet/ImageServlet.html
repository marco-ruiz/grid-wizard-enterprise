17  package org.gwe.rse.servlet;
18  
19  import java.io.File;
20  import java.io.FileInputStream;
21  import java.io.IOException;
22  import java.io.OutputStream;
23  
24  import javax.servlet.ServletContext;
25  import javax.servlet.http.HttpServlet;
26  import javax.servlet.http.HttpServletRequest;
27  import javax.servlet.http.HttpServletResponse;
28  
29  /**
30   * @author Marco Ruiz
31   * @since Feb 27, 2009
32   */
33  public class ImageServlet extends HttpServlet {
34  	
35      private String imageNotFoundFileName;
36  	public static final String URI = "/imageServ";
37  
38  	public ImageServlet(String imageNotFoundFileName) {
39      	this.imageNotFoundFileName = imageNotFoundFileName;
40      }
41  
42  	public void doGet(HttpServletRequest request, HttpServletResponse response) throws IOException {
43  		String filename = request.getParameter("filename");
44      	
45          FileInputStream in = null; 
46          try {
47          	in = getFileInputStream(response, filename);
48          } catch (Exception e) {
49          	in = getFileInputStream(response, imageNotFoundFileName);
50          }
51  
52          OutputStream out = response.getOutputStream();
53      
54          // Copy the contents of the file to the output stream
55          byte[] buf = new byte[1024];
56          int count = 0;
57          while ((count = in.read(buf)) >= 0) 
58          	out.write(buf, 0, count);
59  
60          in.close();
61          out.close();
62      }
63  
64  	private FileInputStream getFileInputStream(HttpServletResponse response, String filename) throws IOException {
65  	    // Get the absolute path of the image
66          ServletContext ctx = getServletContext();
67      
68          // Get the MIME type of the image
69          String mimeType = ctx.getMimeType(filename);
70          if (mimeType == null) {
71              String errorMsg = "Could not get MIME type of " + filename;
72  			ctx.log(errorMsg);
73              response.setStatus(HttpServletResponse.SC_INTERNAL_SERVER_ERROR);
74              throw new IOException(errorMsg);
75          }
76      
77          // Set content type
78          response.setContentType(mimeType);
79      
80          // Set content size
81          File file = new File(filename);
82          response.setContentLength((int)file.length());
83      
84          return new FileInputStream(file);
85      }
86  }
