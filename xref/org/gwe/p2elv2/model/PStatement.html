62  	public void inlineConstants() {
63  		List<PVariable> inlineVars = new ArrayList<PVariable>();
64  		for (PVariable var : vars) 
65  			if (var.isVarConstSingleValue()) inlineVars.add(var);
66  		
67  		for (PVariable inlineVar : inlineVars) {
68  			Map<String, Object> constantModel = new HashMap<String, Object>();
69  			List<String> params = inlineVar.getFunctionInvocation().getParams();
70  			String constVal = params.size() > 0 ? params.get(0) : "";
71  			constantModel.put(inlineVar.getName(), constVal);
72  			setTemplate(VelocityUtils.merge(constantModel, getTemplate()));
73  			
74  			for (PVariable var : vars) {
75  				List<String> inlinedParams = new ArrayList<String>();
76  				for (String param : var.getFunctionInvocation().getParams())
77  					inlinedParams.add(VelocityUtils.merge(constantModel, param));
78  				var.getFunctionInvocation().setParams(inlinedParams);
79  	        }
80          }
81  		
82  		vars.removeAll(inlineVars);
83  	}

85  	public List<PVariable> getVarsSortedByDependencies() throws P2ELFunctionNotSupported, P2ELDependentVariableNotResolvedException {
86  	    List<PVariable> nonRuntimeVars = new ArrayList<PVariable>();
87  	    List<PVariable> runtimeVars = new ArrayList<PVariable>();
88  	    for (PVariable var : groupVariables()) {
89  	    	boolean varIsRuntime = var.isRuntime() || var.isDependentOnSystemVars() || var.isDependentOnRuntimeVars();
90  			List<PVariable> dest = varIsRuntime ? runtimeVars : nonRuntimeVars;
91  	    	dest.add(var);
92          }
93  	    
94  	    List<PVariable> result = new ArrayList<PVariable>();
95  
96  	    // Sort as many non-runtime variables as possible
97  	    try {
98  	    	sortVars(nonRuntimeVars, result);
99  	    } catch(P2ELDependentVariableNotResolvedException e) {
100 		    // If some left over make sure non of them are multi-value (expand to multiple process invocations)
101 	    	// In case they all are just proceed since they could be resolved later with runtime single values.
102 	    	for (PVariable unresolvedVar : nonRuntimeVars)
103 	            if (!unresolvedVar.isSingleValue()) throw e;
104 		    runtimeVars.addAll(nonRuntimeVars);
105 	    }
106 
107 	    // Sort remaining runtime variables
108 	    try {
109 	        sortVars(runtimeVars, result, PVariable.SYS_VAR_PREFIX);
110 	    } catch(P2ELDependentVariableNotResolvedException e) {}
111 	    sortVars(runtimeVars, result, PVariable.RUN_VAR_PREFIX);
112         
113         return result;
114     }

125 	private void sortVars(List<PVariable> unresolved, List<PVariable> resolved, String... resolvedGroups) throws P2ELDependentVariableNotResolvedException {
126 		while (!unresolved.isEmpty()) {
127 			boolean resolvedVar = false;
128 			for (int idx = 0; idx < unresolved.size(); idx++) {
129 				if (unresolved.get(idx).isVarResolvedBy(resolved, resolvedGroups)) {
130 					resolved.add(unresolved.remove(idx));
131 					resolvedVar = true;
132 					break;
133 				}
134             }
135 			if (!resolvedVar) {
136 	            PVariable var = unresolved.get(0);
137 	            throw new P2ELDependentVariableNotResolvedException(var.getVarDependencyNames(), var);
138             }
139 		}
140     }

142 	private Collection<PVariableArray> groupVariables() {
143 		Map<String, PVariableArray> arrayVars = new HashMap<String, PVariableArray>();
144 		for (PVariable var : vars) {
145 			String varName = var.getName();
146 			PVariableArray parentArray = arrayVars.get(varName);
147 			if (parentArray == null) {
148 				parentArray = new PVariableArray(varName);
149 				arrayVars.put(varName, parentArray);
150 			}
151 			parentArray.addVar(var);
152         }
153 
154 		// Add all grouped variables 
155 	    return arrayVars.values();
156     }
