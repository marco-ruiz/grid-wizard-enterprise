79  	private void structureDocumentation() {
80  	    structureDocumentation.put(TAG_DESCRIPTION, "Not Available.");
81      	String lastTag = null;
82      	for (PDocumentationLine line : documentation) {
83      		lastTag = resolveLastTag(line, lastTag);
84      		if (lastTag != null) {
85      			String prefix = structureDocumentation.get(lastTag);
86      			structureDocumentation.put(lastTag, prefix + " " + line.getContent());
87      		}
88          }
89      }
90  
91  	private String resolveLastTag(PDocumentationLine line, String lastToken) {
92  	    String lineTag = line.getTag();
93  	    if (lineTag != null && !lineTag.equals("")) {
94  	    	lastToken = lineTag;
95  	    	structureDocumentation.put(lastToken, "");
96  	    }
97  	    return lastToken;
98      }

140 	public String evaluateContent(String preffix, List<String> args) {
141 		return contextualize(preffix, substituteParameters(args));
142 	}

144 	private String substituteParameters(List<String> args) {
145 	    String resultContent = content;
146 		for (int idx = 0; idx < params.size(); idx++) { 
147 			String value = (idx < args.size()) ? args.get(idx) : "";
148 			resultContent = resultContent.replace(params.get(idx).asParam(), value);
149 		}
150 		
151 	    return resultContent;
152     }
153 
154 	private String contextualize(String preffix, String params) {
155 		params = params.replaceAll("\\$\\$\\{_}", "\\$\\{" + preffix + "}");
156 	    return params.replaceAll("\\$\\$\\{", "\\$\\{" + preffix + "_");
157     }
