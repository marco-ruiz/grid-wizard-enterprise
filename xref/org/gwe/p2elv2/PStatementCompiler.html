31   * @author Marco Ruiz

34  public class PStatementCompiler {
35  	
36  	private PStatement stmt;
37  	private PStatementContext ctx;
38  	
39  	private Map<PVariable, Map<PPermutation, PVarValueSpace>> cachedValueSpaceSet = 
40  		new HashMap<PVariable, Map<PPermutation, PVarValueSpace>>();
41  
42  	private List<PVariable> varsSorted;
43  	
44  	public PStatementCompiler(PStatement stmt, String workspace, KeyStore keys, FilesStager stager) throws P2ELFunctionNotSupported, P2ELDependentVariableNotResolvedException {
45  		this(stmt, new PStatementContext(workspace, keys, stager));
46      }
47  
48  	public PStatementCompiler(PStatement stmt, PStatementContext ctx) throws P2ELFunctionNotSupported, P2ELDependentVariableNotResolvedException {
49  		setStmt(stmt);
50  		this.ctx = ctx;
51      }
52  
53  	public PStatement getStmt() {
54      	return stmt;
55      }
56  
57  	public void setStmt(PStatement stmt) throws P2ELFunctionNotSupported, P2ELDependentVariableNotResolvedException {
58      	this.stmt = stmt;
59      	this.varsSorted = stmt.getVarsSortedByDependencies();
60          for (PVariable var : varsSorted)
61          	cachedValueSpaceSet.put(var, new HashMap<PPermutation, PVarValueSpace>());
62      }
63  
64  	public List<PPermutation> compile() throws P2ELDependentVariableNotResolvedException, P2ELMultiValueVarDependentOnRuntimeVarException, P2ELFunctionNotSupported {
65  		List<PPermutation> initialPermutations = new ArrayList<PPermutation>();
66  		initialPermutations.add(new PPermutation());
67  
68  		Set<PVariable> varsPermutated = initialPermutations.get(0).keySet();
69  		
70  		// Skip the already compiled vars
71  		int index = 0;
72          for (; index < varsSorted.size(); index++) {
73      		if (!varsPermutated.contains(varsSorted.get(index)))
74      			break;
75          }
76          
77  		// Compile the remaining vars up to the runtime ones (if precompiling)
78          for (; index < varsSorted.size(); index++) {
79          	PVariable var = varsSorted.get(index);
80  			if (var.isRuntime() || var.isDependentOnSystemVars()) 
81  				break;
82          	initialPermutations = permutateWithVar(var, initialPermutations);
83          }
84          
85  		// Store remaining 'un-compiled' vars in each permutation as vars pending compilation
86          for (; index < varsSorted.size(); index++) {
87          	PVariable var = varsSorted.get(index);
88      		if (!var.isSingleValue()) throw new P2ELMultiValueVarDependentOnRuntimeVarException(var);
89          	for (PPermutation perm : initialPermutations) perm.addPendingVar(var);
90          }
91          
92          return initialPermutations;
93  	}
94  	
95  	private List<PPermutation> permutateWithVar(PVariable var, List<PPermutation> parentPermutations) throws P2ELDependentVariableNotResolvedException, P2ELFunctionNotSupported {
96  	    List<PPermutation> currPerms = new ArrayList<PPermutation>();
97  	    for (PPermutation parentPerm : parentPermutations)
98              for (PVarValue value : getValueSpace(var, parentPerm)) 
99              	currPerms.add(parentPerm.cloneAddingVar(var, value));
100 
101 	    return currPerms;
102     }
103 	
104 	private PVarValueSpace getValueSpace(PVariable var, PPermutation parentPerm) throws P2ELDependentVariableNotResolvedException, P2ELFunctionNotSupported {
105 	    PPermutation permKey = parentPerm.cloneOnlyWithVars(var.getVarDependencyNames());
106 	    Map<PPermutation, PVarValueSpace> varValueSpaceCache = cachedValueSpaceSet.get(var);
107 	    PVarValueSpace valueSpace = varValueSpaceCache.get(permKey);
108 	    if (valueSpace == null) {
109 	        valueSpace = var.generateValueSpace(permKey, ctx);
110 	    	varValueSpaceCache.put(permKey, valueSpace);
111 	    }
112 	    return valueSpace;
113     }
114 
115 	public List<String> createStatements(List<PPermutation> permutations) throws P2ELDependentVariableNotResolvedException, P2ELMultiValueVarDependentOnRuntimeVarException {
116 		List<String> result = new ArrayList<String>();
117 		for (PPermutation perm : permutations) result.add(perm.merge(stmt.getTemplate()) + "\n");
118 		return result;
119 	}
120 }
