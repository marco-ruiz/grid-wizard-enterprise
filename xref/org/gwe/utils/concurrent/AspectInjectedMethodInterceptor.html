17  package org.gwe.utils.concurrent;
18  
19  import org.aopalliance.intercept.MethodInterceptor;
20  import org.aopalliance.intercept.MethodInvocation;
21  import org.apache.commons.logging.Log;
22  import org.apache.commons.logging.LogFactory;
23  
24  /**
25   * @author Marco Ruiz
26   * @since Jul 2, 2008
27   */
28  class AspectInjectedMethodInterceptor<PROXIED_TYPE> implements MethodInterceptor {
29  
30  	private static Log log = LogFactory.getLog(AspectInjectedMethodInterceptor.class);
31  	
32      private final Class<PROXIED_TYPE> proxiedClass;
33      
34      AspectInjectedMethodInterceptor(Class<PROXIED_TYPE> proxiedClass) {
35  		this.proxiedClass = proxiedClass;
36      }
37      
38      public Object invoke(MethodInvocation invocation) throws Throwable {
39  		return invokeBareMethod(invocation);
40      }
41      
42  	protected Object invokeBareMethod(MethodInvocation invocation) throws Throwable {
43      	Object id = invocation.getArguments()[0];
44      	String methodName = invocation.getMethod().getName();
45          log.info("AOP for " + proxiedClass + ". Invoking '" + methodName + "'  for entity '" + id + "' with arguments (" + argsToString(invocation) + ") ...");
46          Object result = invocation.proceed();
47      	log.info("AOP for " + proxiedClass + ". '" + methodName + "' invoked for entity '" + id + "' ! Return value: \n" + result);
48          return result;
49      }
50  
51  	private String argsToString(MethodInvocation invocation) {
52          String args = "";
53          Object[] array = invocation.getArguments();
54          for (int idx = 1; idx < array.length; idx++) args += array[idx] + ";";
55          return args;
56      }
57  }
