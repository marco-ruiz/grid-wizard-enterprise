40  	public static String[] removeArgs(String[] args, int start, int count) {
41  		String[] trimmedArgs = new String[args.length - count];
42  		for (int idx = 0; idx < start; idx++) trimmedArgs[idx] = args[idx];
43  		for (int idx = start + count; idx < args.length; idx++) trimmedArgs[idx - count] = args[idx];
44  		return trimmedArgs;

51  	public static List<String> split(String whole, String prefixToken) {
52  		List<String> result = new ArrayList<String>();
53  		int startIdx = 0;
54  		int endIdx;
55  		while ((endIdx = whole.indexOf(prefixToken, startIdx)) != -1) {
56  			if (endIdx != 0) result.add(whole.substring(startIdx, endIdx));
57  			startIdx = endIdx + prefixToken.length();
58  		}
59  		result.add(whole.substring(startIdx));
60  		
61  		return result;
62  	}

68  	public static String[] splitBoundaries(String whole, String startToken, String endToken, String endTokenCancelator) {
69  		int startIndex = whole.indexOf(startToken);
70  		if (startIndex == -1) return null;
71  		String result = whole.substring(startIndex + startToken.length());
72  		int endIndex = getCorrespondingEndTokenIndex(result, endToken, endTokenCancelator);
73  		if (endIndex == -1) return null;
74  		
75          // [SUFIX][START_TOKEN][ENCLOSED_PIECE][END_TOKEN][PREFIX]
76  		return new String[]{ whole.substring(0, startIndex), startToken, result.substring(0, endIndex), endToken, result.substring(endIndex + endToken.length())};
77  	}
78  
79  	private static int getCorrespondingEndTokenIndex(String result, String endToken, String endTokenCancelator) {
80  		if (endTokenCancelator == null) return result.indexOf(endToken); 
81  		
82  	    int endIndex = 0;
83  		int endTokensCount = 0;
84  		while (true) {
85  			endIndex = result.indexOf(endToken, endIndex);
86  			if (endIndex == -1) return -1;
87  			endTokensCount++;
88  			int cancelatorsCount = getTokenCounts(result.substring(0, endIndex), endTokenCancelator);
89  			if (cancelatorsCount == endTokensCount - 1) return endIndex;
90  			endIndex += endTokenCancelator.length();
91  		}
92      }
93  	
94  	private static int getTokenCounts(String result, String endTokenCancelator) {
95  		int count = 0;
96  		int index = 0;
97  		while (true) {
98  			index = result.indexOf(endTokenCancelator, index);
99  			if (index == -1) return count;
100 			index += endTokenCancelator.length();
101 			count++;
102 		}
103     }
