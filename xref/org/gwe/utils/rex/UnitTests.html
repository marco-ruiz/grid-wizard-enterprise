52  @REXConfig4Class(rexPieces={"vars", "template"})
53  class Statement {
54  	@REXConfig4ListElement(min=0)
55  	public List<VarDefinition> vars;
56  	@REXConfig4String(pattern=@REXConfig4Field(field=".*"))
57  	public String template;
58  	public List<VarDefinition> getVars() { return vars; }
59  	public void setVars(List<VarDefinition> vars) { this.vars = vars; }
60  	public String getTemplate() { return template; }
61  	public void setTemplate(String template) { this.template = template; }
62  	public String toString() { return vars.toString() + " onto " + template; }
63  }

66  @REXConfig4Class(rexPieces={"\\s*", "declaration", "\\s*=\\s*", "value|functionInvocation", "\\s*"})
67  class VarDefinition {
68  	public VarDeclaration declaration;
69  	public VarValue value;
70  	public FunctionInvocation functionInvocation;
71  	public VarDeclaration getDeclaration() { return declaration; }
72  	public void setDeclaration(VarDeclaration declaration) { this.declaration = declaration; }
73  	public VarValue getValue() { return value; }
74  	public void setValue(VarValue value) { this.value = value; }
75  	public String toString() { return declaration.toString() + " === " + value + " OR " + functionInvocation; }
76  	public FunctionInvocation getFunctionInvocation() { return functionInvocation; }
77  	public void setFunctionInvocation(FunctionInvocation functionInvocation) { this.functionInvocation = functionInvocation; }
78  }

80  @REXConfig4Class(rexPieces={"\\$\\{", "id", "dims", "\\}"})
81  class VarDeclaration {
82  	@REXConfig4String(pattern=@REXConfig4Field(field="[a-zA-Z]\\w*"))
83  	public String id;
84  	@REXConfig4String(optional=true, pattern=@REXConfig4Field(prefix="\\(", suffix="\\)"))
85  	@REXConfig4ListElement(pattern=@REXConfig4Field(field="[\\w]*", suffix="[,]?"), min=1)
86  	public List<String> dims;
87  	public String getId() { return id; }
88  	public void setId(String id) { this.id = id; }
89  	public List<String> getDims() { return dims; }
90  	public void setDims(List<String> dims) { this.dims = dims; }
91  	public String toString() { return id + " -> " + dims; }
92  }

103 @REXConfig4Class(rexPieces={"f:", "functionName", "\\(", "params", "\\)"})
104 class FunctionInvocation {
105 	@REXConfig4String(pattern=@REXConfig4Field(field="[a-zA-Z]\\w*"))
106 	public String functionName;
107 	@REXConfig4String()
108 	@REXConfig4ListElement(pattern=@REXConfig4Field(field="[^\\(\\),]*", suffix="[,]?"), min=1)
109 	public List<String> params;
110 	public String getFunctionName() { return functionName; }
111 	public void setFunctionName(String functionName) { this.functionName = functionName; }
112 	public List<String> getParams() { return params; }
113 	public void setParams(List<String> params) { this.params = params; }
114 	public String toString() { return functionName + "<" + params + ">"; }
