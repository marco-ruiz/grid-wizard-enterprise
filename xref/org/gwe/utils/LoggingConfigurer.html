17  package org.gwe.utils;
18  
19  import java.io.IOException;
20  import java.util.Map;
21  import java.util.Set;
22  
23  import org.apache.log4j.ConsoleAppender;
24  import org.apache.log4j.FileAppender;
25  import org.apache.log4j.Layout;
26  import org.apache.log4j.Level;
27  import org.apache.log4j.Logger;
28  import org.apache.log4j.PatternLayout;
29  
30  /**
31   * @author Marco Ruiz
32   * @since Nov 15, 2007
33   */
34  public class LoggingConfigurer {
35  
36  	private Map<String, String> levels;
37  	private String pattern;
38  	private boolean addConsoleAppender;
39  	private Set<String> logFiles;
40  	
41  	public void init() {
42  		for (String loggerName : levels.keySet()) {
43  			String levelName = levels.get(loggerName);
44  			try {
45  				Level level = Level.toLevel(levelName);
46  				Logger.getLogger(loggerName).setLevel(level);
47  			} catch (IllegalArgumentException e) {
48  				System.err.println("WARNING: Unable to parse '" + levelName + "' as a org.apache.log4j.Level for logger "
49  						+ loggerName + "; ignoring...");
50  			}
51  		}
52  
53  		Layout layoutObj = new PatternLayout(pattern);
54  
55  		if (addConsoleAppender) 
56  			Logger.getRootLogger().addAppender(new ConsoleAppender(layoutObj));
57  
58  		for (String filename : logFiles)
59  			try {
60  				Logger.getRootLogger().addAppender(new FileAppender(layoutObj, filename));
61  			} catch (IOException e) {
62  				System.err.println("WARNING: Unable to create file appender for log file '" + filename + "'; ignoring...");
63  			}
64  	}
65  
66  	public void setLevels(Map<String, String> props) {
67  		this.levels = props;
68  	}
69  
70  	public void setPattern(String pattern) {
71  		this.pattern = pattern;
72  	}
73  
74  	public void setAddConsoleAppender(boolean addConsoleAppender) {
75  		this.addConsoleAppender = addConsoleAppender;
76  	}
77  
78  	public void setLogFiles(Set<String> logFiles) {
79  		this.logFiles = logFiles;
80  	}
