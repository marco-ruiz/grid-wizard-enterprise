93  			BufferedReader rdr = new BufferedReader(new InputStreamReader(p.getInputStream()));
94  			for (int read = 0; read > -1; read = rdr.read(cbuf)) xmlSB.append(new String(cbuf).substring(0, read));
95  			return xmlSB.toString();
96  		} catch (IOException e) {
