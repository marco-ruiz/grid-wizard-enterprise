17  package org.gwe.integration.slicer;
18  
19  import java.io.File;
20  import java.io.IOException;
21  
22  import org.gwe.utils.IOUtils;
23  
24  /**
25   * @author Marco Ruiz
26   * @since Dec 18, 2007
27   */
28  public class GWEEnabledCLMCreatorApp {
29  
30  	private static final String SCRIPT_PREFFIX = "gweCLMP-";
31  
32  	public static void main(String[] args) throws InstantiationException, IllegalAccessException, ClassNotFoundException {
33  
34  		String pluginsDir = System.getenv(AbstractCLMProxyApp.ENV_SLICER_HOME) + "/lib/Slicer3/Plugins";
35  		
36  		File pluginsDirFile = new File(pluginsDir);
37  		
38  		// Clean up any previous installation for the same version
39  	    for (File file : pluginsDirFile.listFiles()) 
40  	    	if (file.getName().startsWith(SCRIPT_PREFFIX)) file.delete();
41  	    
42  	    // Install new CLM proxies
43  		String scriptPrefix = createScriptPrefix();
44  	    for (File file : pluginsDirFile.listFiles()) {
45  	    	if (file.isFile() && !file.getName().startsWith("gwe") && !file.getName().endsWith(".log")) {
46  	    		System.out.print("Generating CLMP for " + file.getName() + "...");
47  	    		String filename = IOUtils.concatenatePaths(pluginsDir, SCRIPT_PREFFIX + file.getName() + ".sh");
48  				File clmGWEProxy = new File(filename);
49  				try {
50  					IOUtils.createLocalExecutableFile(clmGWEProxy.getAbsolutePath(), scriptPrefix + file.getName() + " $@\n");
51  				} catch (IOException e) {
52  					e.printStackTrace();
53  				}
54  	    		System.out.println("... Done!");
55  	    	}
56  		}
57  	}
58  	
59  	private static String createScriptPrefix() {
60  	    String scriptPreffix = "#!/bin/sh\n";
61  		scriptPreffix += "# Script to proxy into a more real slicer module\n\n"; 
62  //		scriptPreffix += "if [ ! -d $GWE_HOME ]; then\n";
63  		scriptPreffix += "export GWE_HOME=`ls -d -t $" + AbstractCLMProxyApp.ENV_SLICER_HOME + "/gwe-*/ | head -1`\n"; 
64  //		scriptPreffix += "fi;\n";
65  //		scriptPreffix += "export GWE_FAT_JAR=`ls $GWE_HOME/lib/gwe-*-fatjar.jar`\n";
66  //		scriptPreffix += "export GWE_CORE_JAR=`ls $GWE_HOME/lib/gwe-*-core.jar`\n"; 
67  //		scriptPreffix += "export GWE_DEP_JAR=`ls $GWE_HOME/lib/gwe-*-dep.jar`\n"; 
68  //		scriptPreffix += "java -Xmx512m -cp $GWE_HOME/conf:$GWE_CORE_JAR:$GWE_DEP_JAR " + GWECLMProxyApp.class.getName() + " ";
69  		scriptPreffix += "$GWE_HOME/bin/gwe-base-script.sh " + GWECLMProxyApp.class.getName() + " ";
70  	    return scriptPreffix;
71      }
72  }
