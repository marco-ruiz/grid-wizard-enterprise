55  	private ChainDesc chain;

57  	public ChainCLMProxyApp(String chainDescriptorFile) throws Exception {
58  		chain = new ChainDesc(chainDescriptorFile);
59  		ExecutableModelParser descParser = new ExecutableModelParser();
60  		for (ChainCLMDesc clmDesc : chain.getCLMs()) {
61  			WrapperCLMProxyApp proxyApp = new WrapperCLMProxyApp(clmDesc.getId());
62  			proxyApp.setXmlModel(descParser.readExecutableModel(proxyApp.generateProxyXML()));
63  			clmApps.add(proxyApp);
64  		}
65  	}

67  	public String generateProxyXML() throws Exception {
68  		ExecutableModel result = new ExecutableModel(createTitle(), "Composite Chains", "", "org/gwe/integration/slicer/chains/chainsSEMTemplate.vm");
69  
70  		for (int moduleIdx = 0; moduleIdx < clmApps.size(); moduleIdx++) {
71  			ExecutableModel currAppModel = clmApps.get(moduleIdx).getXmlModel();
72  			ChainCLMDesc currChainDesc = chain.getCLMs().get(moduleIdx);
73  			String groupTitlePrefix = currAppModel.getTitle() + " - ";
74  	        for (GroupModel groupModel : currAppModel.getGroups()) {
75  				String newLabel = groupTitlePrefix + groupModel.getLabel();
76  				GroupModel currGroup = result.addGroup(newLabel, groupModel.getDescription());
77  	            for (ParameterModel paramModel : groupModel.getParameters()) {
78  	            	// If the argument is not piped from the output of a previous CLM then add it to the XML
79  	            	if (!currChainDesc.getPipedArguments().containsKey(paramModel.getName()))
80  	            		currGroup.addParam(createProxiedParam(currChainDesc.getId(), paramModel));
81  	            }
82              }
83          }
84  		
85  		return result.getXML();
86  	}

88  	private ParameterModel createProxiedParam(String chainLinkId, ParameterModel param) {
89  		String flagsPrefix = createFlagPrefix(chainLinkId);
90  		String valueType = param.getValueType();
91  /*
92  		if ("file".equals(valueType) || "directory".equals(valueType) || "image".equals(valueType))
93  			valueType = "string";
94  */
95  		String label = param.getLabel();
96  		String longFlag = flagsPrefix + param.getLongFlag();
97  		String description = param.getDescription();
98  		String defaultValue = param.getDefaultValue();
99  		
100 		ParameterModel result = new ParameterModel(valueType, label, longFlag, description, defaultValue);
101 		result.setName(longFlag);
102 		result.setFlag(flagsPrefix + param.getFlag());
103 		result.setFiller(param.getFiller(false, true));
104 
105 		return result;
106 	}

117 	    String title = "";
118 		for (AbstractCLMProxyApp proxyApp : clmApps) 
119 			title += proxyApp.getXmlModel().getTitle() + TITLE_SEPARATOR;

121 		return title.substring(0, title.length() - TITLE_SEPARATOR.length());

124 	public String runProxyApp(String[] args) throws Exception {
125 		String result = "";
126 		for (int moduleIdx = 0; moduleIdx < clmApps.size(); moduleIdx++) {
127 			ChainCLMDesc currChainDesc = chain.getCLMs().get(moduleIdx);
128 			List<String> argsList = constructCLMArgs(args, currChainDesc);
129 			Map<String, String> pipedArgsValues = readPipedArgsValues(args, currChainDesc.getPipedArguments());
130 			for (Map.Entry<String, String> clmArg : pipedArgsValues.entrySet()) {
131 				argsList.add("--" + clmArg.getKey());
132 				argsList.add(clmArg.getValue());
133             }
134 			
135 			WrapperCLMProxyApp wrapperApp = (WrapperCLMProxyApp)clmApps.get(moduleIdx);
136 			String[] chainedModuleArgs = fixIndexedArgs(argsList, moduleIdx).toArray(new String[]{});
137 			result += wrapperApp.runProxyApp(chainedModuleArgs) + "\n===\n";
138 //			result += wrapperApp.createOSCommand(chainedModuleArgs) + "\n";
139 		}
140 //		writeScript("#!/bin/sh\n# Autogenerated chain script\n" + result);
141 		return result;
142 	}

152 	private List<String> constructCLMArgs(String[] args, ChainCLMDesc currChainDesc) {
153 	    List<String> result = new ArrayList<String>();
154 	    String flagsPrefix = createFlagPrefix(currChainDesc.getId());
155 	    for (int argIndex = 0; argIndex < args.length; argIndex++) {
156 	    	String newValue = args[argIndex].replace(flagsPrefix, "");
157 	    	if (!newValue.startsWith("--" + ARG_PREFIX)) { 
158 	    		result.add(newValue);
159 	    	} else {
160 	    		if (!args[argIndex + 1].startsWith("--")) argIndex++;
161 	    	}
162 	    }
163 	    
164 	    return result;
165     }

167 	private Map<String, String> readPipedArgsValues(String[] args, Map<String, String> pipedArgsMappings) {
168 	    OptionableAppTemplate invocationTemplate = new OptionableAppTemplate();
169 		Map<String, OptionTemplate> chainedOptions = new HashMap<String, OptionTemplate>();
170         for (String clmArg : pipedArgsMappings.keySet()) {
171         	OptionTemplate option = new OptionTemplate(appendArgPrefix(pipedArgsMappings.get(clmArg)),"");
172         	chainedOptions.put(clmArg, option);
173         	invocationTemplate.addOption(option, null);
174         }
175 	    invocationTemplate.loadArgs(args);
176 	    
177 	    Map<String, String> pipedArgsValues = new HashMap<String, String>();
178 	    for (Map.Entry<String, OptionTemplate> clmArg : chainedOptions.entrySet())
179 	    	pipedArgsValues.put(clmArg.getKey(), invocationTemplate.getArg(clmArg.getValue()));
180 	    
181 	    return pipedArgsValues;
182     }

184 	private List<String> fixIndexedArgs(List<String> argsList, int appIndex) {
185 		Map<String, Integer> indexedArgs = new HashMap<String, Integer>();
186 		for (ParameterModel param : clmApps.get(appIndex).getXmlModel().getParameters())
187 	        if (param.getIndex() != -1) indexedArgs.put(param.getLongFlag(), param.getIndex());
188 		
189 		Map<Integer, String> indexedInvocations = new TreeMap<Integer, String>();
190 		List<String> result = new ArrayList<String>();
191 		for (int argIndex = 0; argIndex < argsList.size(); argIndex++) {
192 			String argInvoc = argsList.get(argIndex);
193 			String arg = argInvoc.replace("-", "");
194 			if (!indexedArgs.containsKey(arg)) {
195 				result.add(argInvoc);
196 			} else {
197 	    		String nextArg = (argsList.size() > argIndex + 1) ? argsList.get(argIndex + 1) : "";
198 	    		if (nextArg == null) nextArg = "";
199 	    		boolean isValueNextArg = !nextArg.startsWith("--");
200 				indexedInvocations.put(indexedArgs.get(arg), isValueNextArg ? nextArg : "");
201 				if (isValueNextArg) argIndex++;
202 			}
203         }
204 		
205 		int count = 0;
206 		for (Map.Entry<Integer, String> entry : indexedInvocations.entrySet()) { 
207 			result.add(count, entry.getValue());
208 			count++;
209 		}
210 		
211 		return result;
212     }
