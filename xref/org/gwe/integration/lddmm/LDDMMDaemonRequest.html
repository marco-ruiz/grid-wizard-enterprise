29  public class LDDMMDaemonRequest extends OSCommandDaemonRequest<LDDMMParams> {
30  	
31  	private Map<String, String> env;
32  	
33  	protected void setup() {
34          // Create environment
35  		env = new HashMap<String, String>();
36  		env.put("X509_USER_PROXY", createFileUnderWorkspace("/x509up", parameters.getX509up()));
37  		env.put("LDDMM_HOME",      parameters.getLddmmHome());
38  		env.put("SERVER_DN",       parameters.getServerDN());
39  		env.put("AUTH_SCHEME",     parameters.getAuthScheme());
40  		env.put("mdasDomainName",  parameters.getMdasDomainName());
41  		env.put("mdasDomainHome",  parameters.getMdasDomainHome());
42  		env.put("srbHost",         parameters.getSrbHost());
43  		env.put("srbPort",         parameters.getSrbPort());
44  		env.put("srbUser",         parameters.getSrbUser());
45  		env.put("defaultResource", parameters.getDefaultResource());
46  
47  		String atlas  = getCommandArgument("-A");
48          String target = getCommandArgument("-T");
49          String outDir = parameters.getDestinationFolder();
50  		if (outDir == null || "".equals(outDir)) outDir = atlas.substring(0, atlas.lastIndexOf('/') + 1) + "results";
51  		outDir = outDir.replaceAll("srbfile:", "");
52  		String atlasName = atlas.substring(atlas.lastIndexOf('/'));
53  		String resultDir = outDir + atlasName.substring(0, atlasName.indexOf('.'));
54          
55          StringBuffer pre = new StringBuffer("#!/bin/bash\n\n");
56          pre.append("cd ..\n");
57          pre.append("chmod -R go=,u=rwX " + workspacePath + "\n\n");
58          pre.append(getEnvironmentAsString());
59          pre.append("ATLAS=" + atlas + "\n");
60          pre.append("TARGET=" + target + "\n");
61          pre.append("ATLASNAME=$(basename $ATLAS | perl -pe's/\\..*//')\n");
62          pre.append("TARGETNAME=$(basename $TARGET | perl -pe's/\\..*//')\n");
63          pre.append("DOMAIN=$(echo $ATLAS | perl -pe's/.*\\/" + parameters.getSrbUser() + "\\.([^-]+-[^\\/]+).*/\\1/')\n");
64          pre.append("export srbUser=" + parameters.getSrbUser() + "\n");
65          pre.append("export mdasDomainName=$DOMAIN\n");
66          pre.append("export mdasDomainHome=$DOMAIN\n");
67          pre.append("export SRB_HOME=/usr/local/bin/srb34/\n");
68          pre.append("Smkdir " + outDir + "\n");
69          pre.append("Smkdir " + resultDir + "\n");
70          pre.append("printenv\n\n");
71   
72          String configPath = createFileUnderWorkspace("/lddmm-config.txt", parameters.getLddmmConfig());
73          pre.append(localizedCmd + " -o srbdir:" + resultDir + " -m -c " + configPath + "\n");
74  
75          localizedCmd = "sh " + createFileUnderWorkspace("/job.sh", pre.toString());
76  	}
77  
78  	protected Map<String, String> getEnvironment() { return env; }
79  	
80  	private String getEnvironmentAsString() {
81  		String result = "";
82  		for (String key : env.keySet()) result += "export " + key + "=" + env.get(key) + "\n";
83  		return result;
84  	}
85  
86  	protected void tearDown() {
87  		
88  	}
89  	
90      // Should never return "null", as this could cause something along the lines of cat | foo, instead of 
91  	// cat blah | foo; one is an infinite loop, the other is just an error.
92      private String getCommandArgument(String argType) {
93          String[] args = localizedCmd.split("\\s+");
94          for(int idx = 0; idx < args.length; idx++) 
95              if (argType.equals(args[idx])) return args[++idx];
96  
97          return "undefined";
98      }
