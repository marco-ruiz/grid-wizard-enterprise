31   * @author Neil Jones

35  public class CondorDriver extends CommandLineResourceManagerDriver {
36  
37  	private static Log log = LogFactory.getLog(CondorDriver.class);
38  	
39  	public static final String CONDOR_SUBMIT_FILE = "agent.condor.submit";
40  
41  	private static String idPrefixToken = "submitted to cluster ";
42  	private static Pattern pattern = Pattern.compile(idPrefixToken + "(\\d+).");
43  	private static Map<String, AllocationPhase> statusFlags = new HashMap<String, AllocationPhase>();
44      static {
45      	statusFlags.put("I", AllocationPhase.SCHEDULED);
46      	statusFlags.put("R", AllocationPhase.ATTAINED);
47      	statusFlags.put("X", AllocationPhase.RELEASED);
48      	statusFlags.put("C", AllocationPhase.RELEASED);
49      }
50      
51  	public CondorDriver() { super("condor_version", "condor_submit", "rm-condor.vm"); }
52  
53      protected String extractIdFromSubmissionOutput(String[] results) {
54  		String all = "";
55  		for (String line : results) all += line;
56  		log.info("Queueing results:\n" + all);
57  		
58  		String subResult = all.substring(all.lastIndexOf(idPrefixToken));
59  		Matcher matcher = pattern.matcher(subResult);
60  		return (matcher.matches()) ? matcher.group(0) : null;
61      }
62  
63      public AllocationPhase killAllocation(AllocationInfo alloc) throws ResourceAllocationException {
64          String condorId = alloc.getSystemPid();
65          if (condorId == null) return AllocationPhase.RELEASED;
66  
67          try {
68  			new ShellCommand("condor_rm " + condorId).runLocally();
69  		} catch (ConnectorException e) {
70  			throw new ResourceAllocationException("Failed to kill allocation '" + alloc.getId() + "'", e);
71  		}
72          return checkAllocation(alloc);
73      }
74  
75      public AllocationPhase checkAllocation(AllocationInfo alloc) throws ResourceAllocationException {
76          String condorId = alloc.getSystemPid();
77          if (condorId == null) return AllocationPhase.RELEASED;
78  
79  		try {
80  	        AllocationPhase res = handleCondorOutput("condor_q", condorId);
81  	        return (res != null) ? res : handleCondorOutput("condor_history", condorId);
82  		} catch (ConnectorException e) {
83  			throw new ResourceAllocationException("Failed to check system level status of allocation '" + alloc.getId() + "'", e);
84  		}
85      }
86  
87      private AllocationPhase handleCondorOutput(String condorCmd, String condorId) throws ConnectorException {
88          String results = new ShellCommand(condorCmd + " " + condorId).runLocally();
89      	AllocationPhase res = null;
90          for(String line : results.split("\n")) {
91              String[] chunks = line.split("\\s+");
92              if (chunks.length >= 9 && chunks[0].equals(condorId + ".0")) 
93              	res = statusFlags.get(chunks[5]);
94          }
95          log.trace("Condor job " + condorId + " has status " + res);
96          return res;
97      }
98  }
