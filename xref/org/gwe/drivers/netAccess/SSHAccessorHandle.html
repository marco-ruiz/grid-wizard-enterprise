34  public class SSHAccessorHandle extends RemoteComputeNodeAccessorHandle {
35  
36      private static Log log = LogFactory.getLog(SSHAccessorHandle.class);
37  
38  	private JSchConnection conn;
39      	
40      public SSHAccessorHandle() throws HandleCreationException {
41      	this(new AccessorLink());
42      }
43      
44      public SSHAccessorHandle(AccessorLink link) throws HandleCreationException {
45      	super(link);
46      	try {
47      		conn = new JSchConnection(link);
48      	} catch (Exception ex) {
49              log.info("Connection failed: aborting", ex);
50              throw new HandleCreationException("Jsch could not established a connection.", ex);
51          }
52      }
53      
54  	public void openTunnel(int localPort, int remotePort) throws HandleOperationException {
55  		try {
56  			// Try to clean it up if it exists
57  			conn.getSessionObj().delPortForwardingL(localPort);
58  		} catch (JSchException e1) {
59  			// Never mind. Maybe it doesn't exist yet
60  		}
61  
62  		try {
63  			conn.getSessionObj().setPortForwardingL(localPort, link.getURI().getHost(), remotePort);
64  			log.info("SSH tunnel created = [localhost:" + localPort + "] >==< [" + link.getURI().getHost() + ":" + remotePort +"]");
65  		} catch (JSchException e) {
66  //			log.debug("Exception thrown while trying to establish tunnel " + tunnel, e);
67  //			throw new HandleCreationException("Could not open a proxy tunnel", e);
68  		}
69  	}
70      
71  	public String runCommand(ShellCommand cmd) throws ConnectorException {
72  		ChannelExec channel = null;
73          try {
74          	//
75              log.debug("Opening execution channel to " + link.getURI());
76              channel = (ChannelExec)conn.getSessionObj().openChannel("exec");
77      		channel.setCommand(cmd.getUnixStyleCmd());
78      		channel.setErrStream(System.err);
79      		InputStream resultStream = channel.getInputStream();
80      		channel.connect(30000);
81      		log.debug("Channel successfully connected");
82      		String res = readExecutionOutput(cmd, channel, resultStream);
83              log.debug("Command results:\n" + res);
84              return res;
85          } catch (JSchException e) {
86              throw new ConnectorException("Error encountered while creating SSH channel to connect to " + link.getURI(), e);
87          } catch (IOException e) {
88              throw new ConnectorException("Error executing command " + cmd.getCmd(), e);
89          } finally {
90      		if (channel != null) channel.disconnect();
91          }
92  	}
93  
94  	private String readExecutionOutput(ShellCommand cmd, final ChannelExec channel, InputStream resultStream) throws IOException {
95  		ExecutionChannelInspector inspector = new ExecutionChannelInspector() {
96  			public boolean isExecutionChannelOpened() {
97  			    if (channel.isClosed()) {
98  			        log.info("Exit status: " + channel.getExitStatus());
99  			        return false;
100 			    }
101 			    try {
102 			        Thread.yield();
103 			    } catch (Exception ex) {} // From the Exec example in jsch.
104 		        return true;
105 		    }
106 		};
107 		return cmd.readExecutionOutput(resultStream, inspector);
108 	}
109 	
110 	public void close() throws ConnectorException {
111 		conn.close();
112         conn = null;
113 	}

116 	protected void finalize() throws Throwable {
117 		try {
118 			close();
119 		} finally {
120 			super.finalize();
121 		}
122 	}
