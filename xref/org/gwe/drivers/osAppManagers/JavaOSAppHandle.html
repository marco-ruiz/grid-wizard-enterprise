17  package org.gwe.drivers.osAppManagers;
18  
19  import org.gwe.utils.IOUtils;
20  
21  /**
22   * @author Marco Ruiz
23   * @since Aug 22, 2007
24   */
25  public class JavaOSAppHandle extends OSAppHandle {
26  	
27  	public static final String FATJAR_SUFFIX = "-fatjar.jar";
28  	private static final int NO_DEBUG_PORT = -1;
29  	
30  	private String mainClassName;
31  	private Object[] args = new Object[]{};
32  	private int debugPort = NO_DEBUG_PORT;
33  	private String[] jars;
34  	
35  	JavaOSAppHandle(String installationPath, String name, String version, String mainClassName) {
36  		this(installationPath, name, version, mainClassName, null);
37  	}
38  	
39  	JavaOSAppHandle(String installationPath, String name, String version, String mainClassName, String... jars) {
40  		super(installationPath, name, version);
41  		this.mainClassName = mainClassName;
42  		this.jars = jars;
43  	}
44  	
45  	public void setDebugPort(int debugPort) {
46  		this.debugPort = debugPort;
47  	}
48  	
49  	public void resetDebugPort() {
50  		this.debugPort = NO_DEBUG_PORT;
51  	}
52  	
53  	public void setArgs(Object... args) {
54  		if (args == null) return;
55  		this.args = args;
56  	}
57  	
58  	public String getClasspath() {
59  		return getFolderFullPath(OSAppFolder.CONFIGURATIONS) + ":" + getFolderFullPath(OSAppFolder.LIBRARIES) + createClasspath();
60  	}
61  	
62  	private String createClasspath() {
63  		String result = "";
64  		if (jars == null) jars = new String[]{getStandardVersionedTemplate(getName() + FATJAR_SUFFIX)};
65  		for (String jar : jars) result += ":" + getJarLocation(jar);
66  		return result;
67  	}
68  	
69  	private String getJarLocation(String jar) {
70  		return IOUtils.createFullFileName(getFolderFullPath(OSAppFolder.LIBRARIES), getVersioned(jar));
71  	}
72  	
73  	public String getLaunchCommand() {
74  		return getLaunchCommand(args);
75  	}
76  	
77  	public String getLaunchCommand(Object... args) {
78  		String argsStr = "";
79  		if (args != null) 
80  			for (Object arg : args) argsStr += "\"" + arg + "\" ";
81  		
82  		String debugParams = (debugPort == NO_DEBUG_PORT) ? "" : " -agentlib:jdwp=transport=dt_socket,server=y,address=" + debugPort;
83  		return "java -Xmx512m " + debugParams + " -cp " + getClasspath() + " " + mainClassName + " " + argsStr;
84  	}
85  	
86  	public JavaOSAppHandle clone() {
87  		JavaOSAppHandle result = new JavaOSAppHandle(installPath, name, version, mainClassName);
88  		result.setDebugPort(debugPort);
89  		return result;
90  	}
91  }
