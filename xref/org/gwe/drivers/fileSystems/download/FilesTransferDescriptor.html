16  	public static String getLocalFileName(String workspace, String remoteFile, String newName) {
17  		if (newName != null) remoteFile = remoteFile.substring(0, remoteFile.lastIndexOf("/") + 1) + newName;
18  		return workspace + LOCALIZED_FS_DIR + remoteFile.replace("://", "/");
19  	}

21  	private String prefixDirection;
22  	private Map<String, String> repo = new HashMap<String, String>();
23  	
24  	public FilesTransferDescriptor(String prefixDirection, String command, String workspace) {
25  		this.prefixDirection = prefixDirection;
26  		String[] args = StringUtils.splitSpaceSeparated(command);
27  		for (int idx = 0; idx < args.length; idx++) {
28  			String currArg = args[idx];
29  			if (currArg.startsWith(prefixDirection)) {
30  				String prefix = prefixDirection + "[";
31  				String newName = (currArg.startsWith(prefix)) ? 
32  						currArg.substring(prefix.length(), currArg.indexOf("]")) : null;
33  				repo.put(currArg, FilesTransferDescriptor.getLocalFileName(workspace, getURI(currArg), newName));
34  			}
35  		}
36  	}
37  	
38  	public Map<String, String> getFileTransfers() {
39  		Map<String, String> result = new HashMap<String, String>();
40  		for (String remoteFile : repo.keySet()) 
41  			result.put(getURI(remoteFile), repo.get(remoteFile));
42  		return result; 
43  	}
44  
45  	private String getURI(String remoteFile) {
46  		return remoteFile.substring(remoteFile.indexOf(':') + 1);
47  	}

50  		Map<String, String> result = new HashMap<String, String>();
51  //		String prefix = " " + prefixDirection;

54  		
55  		return result;
56  	}
