32  	private Set<DownloadHandle> repo = new TreeSet<DownloadHandle>();
33  	private long repoSize;
34  	private int filesDownloadedSinceLastGC = 0;
35  	
36  	public DownloadHandleRepo(double gigabytes) {
37  		this.repoSize = (long)(gigabytes * BYTES_IN_GYGABYTE);
38  		new Thread(this, "BG Service - Virtual Cache FS Garbage Collector").start();
39  	}

41  	public DownloadHandle createHandle(String downloadedFile, Future<Long> future) {
42  		DownloadHandle handle = new DownloadHandle(downloadedFile, future, this);
43  		synchronized (this) {
44  			repo.add(handle);
45  			filesDownloadedSinceLastGC++;
46  		}
47  		return handle;
48  	}

50  	public void startGarbageCollection() {
51  		synchronized (this) { this.notifyAll(); }
52  	}

54  	public void run() {
55  		while (true) {
56  			synchronized (this) {
57  				try {
58  					wait();
59  				} catch (InterruptedException e) {}
60  				garbageCollectAsNeeded();
61  			}
62  		}
63  	}
64  
65  	private void garbageCollectAsNeeded() {
66  		long handleSize;
67  		long memoryCount = repoSize;
68  //		log.debug("Repository before garbage collection:\n" + repo);
69  		Set<DownloadHandle> removedHandles = new HashSet<DownloadHandle>();
70  		for (DownloadHandle handle : repo) {
71  			if (memoryCount > 0) {
72  				handleSize = handle.getSize();
73  				if (handleSize != DownloadHandle.CANNOT_FREE_NOW_SIZE) memoryCount -= handleSize;
74  			} else {
75  				handleSize = handle.garbageCollect();
76  				if (handleSize != DownloadHandle.CANNOT_FREE_NOW_SIZE) {
77  					deleteFileSystemEntry(handle.localFile);
78  					removedHandles.add(handle);
79  				}
80  			}
81  			repo.removeAll(removedHandles);
82  		}
83  //		log.debug("Repository after garbage collection:\n" + repo);
84  		filesDownloadedSinceLastGC = 0;
85  	}
86  
87  	public void disposeDownloadsUnder(String topLevelFolder) {
88  		synchronized (this) {
89  //			log.debug("Repository before disposing order virtual cache FS:\n" + repo);
90  			deleteFileSystemEntry(topLevelFolder);
91  			Set<DownloadHandle> removedHandles = new HashSet<DownloadHandle>();
92  			for (DownloadHandle handle : repo) {
93  				if (handle.localFile != null && handle.localFile.startsWith(topLevelFolder)) {
94  					handle.garbageCollect();
95  					removedHandles.add(handle);
96  				}
97  			}
98  			repo.removeAll(removedHandles);
99  //			log.debug("Repository before disposing order virtual cache FS:\n" + repo);
100 		}
101 	}

108 			// TODO: Ignore for now
