24  import org.apache.commons.logging.Log;

28   * @author Marco Ruiz

31  public class DownloadHandleRepo implements Runnable {
32  
33      private static Log log = LogFactory.getLog(DownloadHandleRepo.class);
34  
35  	private static long BYTES_IN_GYGABYTE = 1024 * 1024 * 1024;
36  	
37  	private static DownloadHandleRepo instance = null;
38  	
39  	// TODO: Move to Spring factory
40  	public static DownloadHandleRepo getInstance() {
41  		if (instance == null) instance = new DownloadHandleRepo(0.1); // 100 MB
42  //		if (instance == null) instance = new DownloadHandleRepo(0.5); // Half a gigabyte
43  		return instance;
44  	}
45  	
46  	private Set<DownloadHandle> repo = new TreeSet<DownloadHandle>();
47  	private long repoSize;
48  	private int filesDownloadedSinceLastGC = 0;
49  	
50  	public DownloadHandleRepo(double gigabytes) {
51  		this.repoSize = (long)(gigabytes * BYTES_IN_GYGABYTE);
52  		new Thread(this, "BG Service - Virtual Cache FS Garbage Collector").start();
53  	}
54  
55  	public DownloadHandle createHandle(String downloadedFile, FilesStager stager, Future<Long> future) {
56  		DownloadHandle handle = new DownloadHandle(downloadedFile, stager, future, this);
57  		synchronized (this) {
58  			repo.add(handle);
59  			filesDownloadedSinceLastGC++;
60  		}
61  		return handle;
62  	}
63  	
64  	public void startGarbageCollection() {
65  		synchronized (this) { this.notifyAll(); }
66  	}
67  
68  	public void run() {
69  		while (true) {
70  			synchronized (this) {
71  				try {
72  					wait();
73  				} catch (InterruptedException e) {}
74  				garbageCollectAsNeeded();
75  			}
76  		}
77  	}
78  
79  	private void garbageCollectAsNeeded() {
80  		long handleSize;
81  		long memoryCount = repoSize;
82  //		log.debug("Repository before garbage collection:\n" + repo);
83  		Set<DownloadHandle> removedHandles = new HashSet<DownloadHandle>();
84  		for (DownloadHandle handle : repo) {
85  			if (memoryCount > 0) {
86  				handleSize = handle.getSize();
87  				if (handleSize != DownloadHandle.CANNOT_FREE_NOW_SIZE) memoryCount -= handleSize;
88  			} else {
89  				handleSize = handle.garbageCollect(true);
90  				if (handleSize != DownloadHandle.CANNOT_FREE_NOW_SIZE) removedHandles.add(handle);
91  			}
92  			repo.removeAll(removedHandles);
93  		}
94  //		log.debug("Repository after garbage collection:\n" + repo);
95  		filesDownloadedSinceLastGC = 0;
96  	}
97  
98  	public void disposeDownloadsUnder(String topLevelFolder) {
99  		synchronized (this) {
100 //			log.debug("Repository before disposing order virtual cache FS:\n" + repo);
101 //			deleteFileSystemEntry(topLevelFolder, account);
102 			Set<DownloadHandle> removedHandles = new HashSet<DownloadHandle>();
103 			for (DownloadHandle handle : repo) {
104 				if (handle.localFile != null && handle.localFile.startsWith(topLevelFolder)) {
105 					handle.garbageCollect(false);
106 					removedHandles.add(handle);
107 				}
108 			}
109 			repo.removeAll(removedHandles);
110 //			log.debug("Repository before disposing order virtual cache FS:\n" + repo);
111 		}
112 	}
113 }
