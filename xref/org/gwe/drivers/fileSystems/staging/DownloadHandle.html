17  package org.gwe.drivers.fileSystems.staging;
18  
19  import java.util.concurrent.ExecutionException;
20  import java.util.concurrent.Future;
21  
22  import org.apache.commons.logging.Log;
23  import org.apache.commons.logging.LogFactory;
24  
25  /**
26   * @author Marco Ruiz
27   * @since Feb 20, 2008
28   */
29  class DownloadHandle implements Comparable<DownloadHandle> {
30  
31      private static Log log = LogFactory.getLog(DownloadHandle.class);
32  
33      static long CANNOT_FREE_NOW_SIZE = -1;
34  	
35  	String localFile;
36  	private FilesStager stager;
37  	
38  	private Future<Long> result;
39  	private DownloadHandleRepo repo;
40  	
41  	private long size = CANNOT_FREE_NOW_SIZE;
42  
43  	private Long lastTimeAccessed = System.currentTimeMillis();
44  	private int clients = 0;
45  	private Boolean garbageCollected = false;
46  
47  	DownloadHandle(String downloadedFile, FilesStager stager, Future<Long> future, DownloadHandleRepo repo) {
48  		this.localFile = downloadedFile;
49  		this.stager = stager;
50  		this.result = future;
51  		this.repo = repo;
52  	}
53  
54  	public String getLocalFileWhenReady() throws InterruptedException, ExecutionException {
55  		size = result.get();
56  		return localFile;
57  	}
58  
59  	long garbageCollect(boolean disposeAssociatedFile) {
60  		// Unnecessary synchronization since it will always be invoke from recycling thread which already owns this lock
61  		synchronized (repo) {
62  			if (clients > 0) return CANNOT_FREE_NOW_SIZE;
63  			if (disposeAssociatedFile) stager.cleanUp(localFile);
64  			log.info("Garbage collecting download: " + this.toString());
65  			garbageCollected = true;
66  			return size;
67  		}
68  	}
69  	
70  	public boolean lock() {
71  		synchronized (repo) {
72  			if (garbageCollected) return false;
73  			clients++;
74  			return true;
75  		}
76  	}
77  
78  	public void unlock() {
79  		synchronized (repo) {
80  			clients--;
81  			if (clients == 0) lastTimeAccessed = System.currentTimeMillis();
82  		}
83  	}
84  
85  	public int compareTo(DownloadHandle other) {
86  		return (this.clients > 0) ? 1 : (int)(this.lastTimeAccessed - other.lastTimeAccessed);
87  	}
88  
89  	public long getSize() {
90  		return size;
91  	}
92  	
93  	public String toString() {
94  		return localFile + "[" + clients + "-" + garbageCollected + "]=" + size + ";" + (System.currentTimeMillis() - lastTimeAccessed) / 1000 + " secs ago";
95  	}
96  }
