52  	public String runPrecompilationPhase(P2ELFunctionInvocationPhase phase, P2ELExpressionContext ctx, 
53  			String params, List<P2ELVariable> vars) throws P2ELSyntaxException {
54  		
55      	if (phase != P2ELFunctionInvocationPhase.EXPAND) return null;
56  		List<String> varReferences = P2ELVariable.extractVarReferences(params);
57  		
58  		if (varReferences.size() == 0 && params.contains("*")) {
59  			String wildcardAnonVarValue = params.substring(params.indexOf("*"), params.lastIndexOf("*") + 1);
60  //			P2ELVariable wildcardAnonVar = new P2ELVariable((P2EL_AUTOGEN_WC_VAR_PREFIX + invocationCount), wildcardAnonVarValue);
61  			P2ELVariable wildcardAnonVar = new P2ELVariable(P2EL_AUTOGEN_WC_VAR_PREFIX + (vars.size() + 1), wildcardAnonVarValue);
62  			vars.add(wildcardAnonVar);
63  			params = params.replace(wildcardAnonVarValue, wildcardAnonVar.getReference());
64  			varReferences.add(wildcardAnonVar.getReference());
65  		}
66  		
67  		if (varReferences.size() == 1 && !params.contains("*")) {
68  			String varRef = varReferences.get(0);
69  			P2ELVariable varDeclaration = getVarDeclaration(varRef, vars);
70  			if (varDeclaration != null && varDeclaration.getRanges().size() == 1) {
71  				String wcGlobExp = varDeclaration.getRanges().get(0).getRange();
72  				wcGlobExp = wcGlobExp.substring(1, wcGlobExp.length() - 1); // Remove the quotes
73  				if (wcGlobExp.contains("*")) {
74  					String globFilePath    = params.replace(varRef, wcGlobExp);
75                      String regExpFilePath  = params.replace(varRef, "(" + wcGlobExp.replace(".", "\\.").replace("**", "*").replace("*", ".*") + ")");
76  					List<String> wildcardMatches = resolveWildcardMatches(globFilePath, ctx.getKeys());
77  					varDeclaration.setRanges(createVarRanges(regExpFilePath, wildcardMatches));
78  				}
79  			}
80  		}
81  		
82  		return params;
83  	}

85  	private P2ELVariable getVarDeclaration(String varReference, List<P2ELVariable> vars) {
86  	    for (P2ELVariable var : vars) {
87  	        if (var.getReference().equals(varReference)) return var;
88  	    }
89  	    return null;
90      }

92  	private List<P2ELRange> createVarRanges(String regExpFilePath, List<String> matches) throws P2ELSyntaxException {
93  		regExpFilePath = extractFilePath(regExpFilePath);
94  	    List<P2ELRange> ranges = new ArrayList<P2ELRange>();
95          Pattern pattern = Pattern.compile(regExpFilePath);
96          for (String match : matches) {
97          	Matcher matcher = pattern.matcher(match);
98          	if (matcher.matches()) 
99          		ranges.add(P2ELRange.createRange(matcher.group(1)));
100         }
101         return ranges;
102     }

104 	private static String extractFilePath(String fileURI) throws P2ELSyntaxException {
105 		try {
106 	        URI uriObj = new URI(IOUtils.getFileBase(fileURI));
107 	        int startIndex = fileURI.indexOf(uriObj.getPath());
108 	        return fileURI.substring(startIndex);
109         } catch (URISyntaxException e) {
110         	throw new P2ELSyntaxException("Syntax error found in wildcard file path '" + fileURI + "'");
111         }

119 	private List<String> resolveWildcardMatches(String globFilePath, KeyStore keys) throws P2ELSyntaxException {
120 	    try {
121 			URILink link = (keys != null) ? keys.generateURILink(globFilePath) : URILink.createLocalLink(new URI(globFilePath));
122 			return GWEAppContext.getGridFileSystem().findWildcardMatches(link);
123         } catch (URISyntaxException e) {
124         	throw new P2ELSyntaxException("", e);
125         } catch (HandleOperationException e) {
126         	throw new P2ELSyntaxException("", e);
127         }
128     }
