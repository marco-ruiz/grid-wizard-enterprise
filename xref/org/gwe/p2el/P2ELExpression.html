28  public class P2ELExpression {
29  
30  	private String expression;
31  	private P2ELExpressionContext context;
32  
33  	private String exprSufix;
34  	private List<P2ELFunctionInvocation> invocations = new ArrayList<P2ELFunctionInvocation>();
35  	
36  	P2ELExpression(P2ELExpressionContext ctx, String expr, List<P2ELFunction> functions) throws P2ELSyntaxException {
37  		this.expression = expr;
38  		this.context = ctx;
39  
40  		String prefix = "";
41          String remainder = expr;
42          while (true) {
43          	String[] pieces = StringUtils.splitBoundaries(remainder, P2ELFunctionInvocation.FUNCTION_PREFIX, "(");
44          	if (pieces == null) break;
45          	String functionName = pieces[2];
46          	P2ELFunction<?> function = getFunction(functionName, functions);
47          	if (function == null) {
48          		prefix   += pieces[0] + pieces[1];
49          		remainder = pieces[2] + pieces[3] + pieces[4];
50          	} else {
51          		// Identify start and end indexes of function invocation
52          		String[] invocationPieces = StringUtils.splitBoundaries(remainder, P2ELFunctionInvocation.FUNCTION_PREFIX + functionName + "(", ")", "(");
53          		if (invocationPieces == null) throw new P2ELSyntaxException("Function invocation missing closing parenthesis");
54          		P2ELExpression paramsExpr = new P2ELExpression(ctx, invocationPieces[2], functions);
55  				invocations.add(function.createInvocation(prefix + invocationPieces[0], paramsExpr));
56  				prefix = "";
57          		remainder = invocationPieces[4];
58          	}
59          }
60  		this.exprSufix = prefix + remainder;
61  	}
62  	
63  	public P2ELExpressionContext getContext() {
64      	return context;
65      }
66  	
67  	private P2ELFunction getFunction(String id, List<P2ELFunction> functions) {
68  		for (P2ELFunction function : functions) 
69  			if (function.getId().equals(id)) return function;
70  		
71  	    return null;
72      }
73  	
74  	public String runPrecompilationPhase(P2ELFunctionInvocationPhase phase, List<P2ELVariable> exprVars) throws P2ELSyntaxException {
75  		return runCommandPhase(phase, exprVars);
76  	}
77  	
78  	public String compile() throws P2ELSyntaxException {
79  		return runCommandPhase(P2ELFunctionInvocationPhase.COMPILE);
80      }
81  	
82  	private String runCommandPhase(P2ELFunctionInvocationPhase phase, Object... phaseArgs) throws P2ELSyntaxException {
83  		try {
84  			for (P2ELFunctionInvocation invoc : invocations) {
85  	            switch (phase) {
86  	            	case INTERPRET:  
87  	            	case EXPAND:  
88  	            	case PREEVALUATE:  
89  	            					invoc.runPrecompilationPhase(phase, context, (List<P2ELVariable>) phaseArgs[0]); break;
90  	            	case COMPILE:   invoc.compile (context); break;
91  	            }
92  			}
93  			return toString();
94          } catch (P2ELSyntaxException e) {
95          	throw e;
96          } catch (Exception e) {
97          	throw new P2ELSyntaxException("An error was encountered while trying to " + phase.getName() + " the P2EL statement " + expression, e);
98          }
99  	}

102 //		return getCompilationModels(P2ELFunction.class);
103 		return getCompilationModels(null);
104     }

106 	public List<Object> getCompilationModels(Class<? extends P2ELFunction> functionClass) {
107 		List<Object> result = new ArrayList<Object>();
108 		Object compModel;
109 		for (P2ELFunctionInvocation invoc : invocations) {
110 			if (functionClass == null || functionClass.isAssignableFrom(invoc.getFunctionClass())) {
111 				compModel = invoc.getCompilationModel();
112 				if (compModel != null) result.add(compModel);
113 			}
114 		}
115 
116 		return result;
117     }
118 	
119 	public String toString() {
120 		StringBuffer result = new StringBuffer();
121 		for (P2ELFunctionInvocation invoc : invocations) result.append(invoc.toString());
122 		return result.append(exprSufix).toString();
123 	}
