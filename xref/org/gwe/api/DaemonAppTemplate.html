17  package org.gwe.api;
18  
19  import org.gwe.drivers.osAppManagers.OSAppFolder;
20  import org.gwe.persistence.model.HeadResourceInfo;
21  import org.gwe.persistence.model.PlatformType;
22  import org.gwe.utils.cmd.OptionTemplate;
23  import org.gwe.utils.cmd.OptionableAppTemplate;
24  
25  /**
26   * @author Marco Ruiz
27   * @since Dec 12, 2007
28   */
29  public class DaemonAppTemplate extends OptionableAppTemplate {
30  
31  	private static String CONFIG_MODE_NO  = "no";
32  	private static String CONFIG_MODE_YES = "yes";
33  	
34  	private final OptionTemplate OPT_HOST = new OptionTemplate('H', "host", "Host address of this daemon as referenced by clients");
35  	private final OptionTemplate OPT_PORT = new OptionTemplate('p', "port", "Port where the RMI registry daemon will run", "1099");
36  	private final OptionTemplate OPT_DEST = new OptionTemplate('d', "dest", "Root directory where to install GWE applications");
37  
38  	private final OptionTemplate OPT_NAME = new OptionTemplate('n', "name", "Friendly name of this daemon. To be persisted in the DB");
39  	private final OptionTemplate OPT_AGE  = new OptionTemplate('a', "max_age", "Maximum number of minutes an allocation can remain alive");
40  	private final OptionTemplate OPT_IDLE = new OptionTemplate('i', "max_idle", "Maximum number of minutes an allocation can remain alive in idle state");
41  	private final OptionTemplate OPT_SIZE = new OptionTemplate('s', "size", "Maximum number of simultaneous allocations. A value less than one represent no limit");
42  	
43  	private final OptionTemplate OPT_MODE = new OptionTemplate('c', "config_mode", "Application mode to only run initial installation configuration. By default is not set", CONFIG_MODE_NO);
44  	private final OptionTemplate OPT_EXIT = new OptionTemplate('t', "exit_token", "Token to be output after application completes launching");
45  	
46      public DaemonAppTemplate(OptionTemplate... additionalOptions) { 
47      	addOptions(OPT_HOST, OPT_PORT, OPT_DEST, OPT_NAME, OPT_AGE, OPT_IDLE, OPT_SIZE, OPT_MODE, OPT_EXIT);
48      	addOptions(additionalOptions);
49      }
50  	
51  	public HeadResourceInfo createDaemonInfoUsingArgs() {
52  		HeadResourceInfo daemonInfo = new HeadResourceInfo(getArg(OPT_HOST), getArgAsInt(OPT_PORT));
53  		daemonInfo.setName(getArg(OPT_NAME));
54  		
55  //		Long age = getMinArgAsMillis(OPT_AGE);
56  		Long age = getArgAsLong(OPT_AGE);
57  		if (age != null) daemonInfo.setMaxHijackMillis(age);
58  		
59  //		Long idle = getMinArgAsMillis(OPT_IDLE);
60  		Long idle = getArgAsLong(OPT_IDLE);
61  		if (idle != null) daemonInfo.setMaxIdleMillis(idle);
62  		
63  		Integer size = getArgAsInt(OPT_SIZE);
64  		if (size != null) daemonInfo.setQueueSize(size); 
65  
66  		daemonInfo.setDaemonOSApp(null);
67  		
68  		System.getProperty("os.name");
69  		daemonInfo.setPlatform(PlatformType.detectRunningPlatform());
70  		
71  		return daemonInfo;
72  	}
73  
74  	public String getGWEAppsRoot() {
75  		return getArg(OPT_DEST);
76  	}
77  	
78  	public boolean isConfigMode() {
79  		return !getArg(OPT_MODE).equals(CONFIG_MODE_NO);
80  	}
81  	
82  	public String getExitToken() {
83  		return getArg(OPT_EXIT);
84  	}
85  	
86  	public String[] createAppArgs(boolean configMode, HeadResourceInfo daemonInfo, String exitToken) {
87  		OptionableAppTemplate result = new OptionableAppTemplate();
88  		if (configMode) result.addOption(OPT_MODE, CONFIG_MODE_YES);
89  		result.addOption(OPT_NAME, daemonInfo.getName());
90  		result.addOption(OPT_HOST, daemonInfo.getHost());
91  		result.addOption(OPT_PORT, daemonInfo.getRegistryPort());
92  		result.addOption(OPT_DEST, daemonInfo.getDaemonOSApp().getFolderFullPath(OSAppFolder.ROOT));
93  		result.addOption(OPT_AGE,  daemonInfo.getMaxHijackMillis());
94  		result.addOption(OPT_IDLE, daemonInfo.getMaxIdleMillis());
95  		result.addOption(OPT_SIZE, daemonInfo.getQueueSize());
96  		result.addOption(OPT_EXIT, exitToken);
97  		return result.getInvocationArgs();
98  	}
99  }
