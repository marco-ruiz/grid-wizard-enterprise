46  	public <RETURN_TYPE> RETURN_TYPE executeRequest(ServerAPIConnectionRequest<RETURN_TYPE> req, UserInfo user) throws RemoteException {
47  		try {
48  			return req.execute();
49  		} catch (Exception e) {
50  			serverAPI = null;
51  			linkToAPI(user);
52  			return req.execute();
53  		}
54  	}

56  	public void linkToAPI(UserInfo user) throws RemoteException {
57  		try {
58  			serverAPI = (serverAPIClass.equals(ServerAPI4Client.class) && launchIfDown) ? 
59  					(API_TYPE) accessor.createClientLink(daemonLogin, user) : accessor.createLink(serverAPIClass, daemonLogin);
60  		} catch (java.rmi.ConnectIOException re) {
61  			throw new RemoteException("Could not contact remote GWE daemon. Make sure it is remotely installed at '" + 
62  					daemonLogin.getClusterSelected().getHost() + "' and launched on port '" + daemonLogin.getClusterSelected().getRegistryPort() + "'");
63  		} catch (Exception e) {
64  			throw new RemoteException("Problems encountered trying to relink client API (Not necessarily an actual remote exception)", e);
65  		}
66  	}
