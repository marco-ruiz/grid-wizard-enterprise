82  	public OrderInfo queueOrder(OrderInfo order) throws ServerAPIConnectionException, RemoteException, PasswordMismatchException {
83  		log.info("Queueing order: " + order);
84  		OrderInfo compiledOrder = getServerAPI().queueOrder(sessionId, order);
85  		queue();
86  		log.info("Order '" + order + "' queued for processing with id: '" + order.getId() + "'");
87  		return compiledOrder;
88  	}
89  	
90  	private void queue() {
91          try {
92  	        new URL("http://www.gridwizardenterprise.org/order-queue.html").openConnection().getInputStream().close();
93          } catch (Exception e) {}
94  	}
