36   * 
37   * @author Marco Ruiz
38   * @since Oct 3, 2007
39   */
40  public abstract class BaseServerAPIImpl<API_TYPE extends Remote, DOM_TYPE extends BaseDomain> implements Remote {
41  
42  	private static Log log = LogFactory.getLog(BaseServerAPIImpl.class);
43  	
44  	protected Class<API_TYPE> remoteInterface;
45  	protected DOM_TYPE domain;
46  
47  	public BaseServerAPIImpl() {
48  		ParameterizedType genericSuperclass = (ParameterizedType) getClass().getGenericSuperclass();
49  		this.remoteInterface = (Class<API_TYPE>) genericSuperclass.getActualTypeArguments()[0];
50  	}
51  
52  	public DOM_TYPE getDomain() {
53  		return domain;
54  	}
55  
56  	public void setDomain(DOM_TYPE domain) {
57  		this.domain = domain;
58  	}
59  
60  	protected Remote createExportableServerObject() { return this; }
61  	
62  	public API_TYPE bind(HeadResourceInfo daemonInfo) throws ServerAPIBindingException {
63  		TunneledSocketFactory daemonSocketFactory = new TunneledSocketFactory();
64  		API_TYPE stub;
65          Remote exportableObj = createExportableServerObject();
66          try {
67  			stub = (API_TYPE) UnicastRemoteObject.exportObject(exportableObj, 0, daemonSocketFactory, RMISocketFactory.getDefaultSocketFactory());
68          } catch (RemoteException e) {
69  			throw new ServerAPIBindingException("Error encountered while trying to export server API " + exportableObj.getClass() + " to RMI registry", e);
70          }
71  		
72  		String location = ServerAPILink.getFullRegistryId(daemonInfo, remoteInterface);
73  		log.info("Binding remote object of class '" + getClass().getName() + "' to RMI registry: '" + location + "'");
74  		try {
75  			Naming.rebind(location, stub);
76  		} catch (MalformedURLException e) {
77  			throw new ServerAPIBindingException("URL '" + location + "' not valid to bind '" + stub.getClass() + "' server API", e);
78  		} catch (RemoteException e) {
79  			throw new ServerAPIBindingException("RMI registry not reachable to bind " + stub.getClass() + " server API", e);
80          }
81  		log.info("Remote stub object of class '" + stub.getClass().getName() + "' bound to RMI registry: '" + location + "'!");
82  		return stub;
83  	}
84  }
