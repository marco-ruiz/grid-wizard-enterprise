75  	public ServerAPI4Client createClientLink(ClientAppConfig daemonLogin, UserInfo user) throws Exception {
76  		ServerAPI4Client coordinator;
77  		try {
78  			coordinator = createLink(ServerAPI4Client.class, daemonLogin);
79  		} catch (Exception e1) {
80  			log.debug("Could not connect to remote daemon.", e1);
81  			if (user == null) throw e1;
82  			try {
83  				deployDaemonOSApp(user, daemonLogin);
84  				launchDaemonOSApp(user, daemonLogin);
85  				coordinator = createLink(ServerAPI4Client.class, daemonLogin);
86  			} catch(Exception e2) {
87  				log.fatal("Could not connect to remote daemon. Fatal Error. Exiting.", e2);
88  				throw e2;
89  			}
90  		}
91  		log.info("Cluster daemon connected!");
92  		return coordinator;
93  	}

99  	public boolean deployDaemonOSApp(UserInfo user, ClientAppConfig daemonLogin, String sourceFolder, String destDir) throws Exception {
100 		if (tryToStartDaemon(user, daemonLogin)) return true;
101 		JavaOSApp daemonOSApp = appFactory.createHandle(destDir, OSAppType.DAEMON);
102 		DaemonAppManager daemonAppMgr = new DaemonAppManager(packageMgr, daemonOSApp, daemonLogin);
103 		daemonAppMgr.installDaemon(sourceFolder);
104 		return tryToStartDaemon(user, daemonLogin);
105 	}

111 	public boolean launchDaemonOSApp(UserInfo user, ClientAppConfig daemonLogin, String destDir) throws Exception {
112 		if (tryToStartDaemon(user, daemonLogin)) return true;
113 		JavaOSApp daemonOSApp = appFactory.createHandle(destDir, OSAppType.DAEMON);
114 		DaemonAppManager daemonAppMgr = new DaemonAppManager(packageMgr, daemonOSApp, daemonLogin);
115 		daemonAppMgr.launchDaemon();
116 		return tryToStartDaemon(user, daemonLogin);
117 	}

140 	public <RT extends Remote> RT createLink(Class<RT> remoteClass, HeadResourceInfo headInfo, URILink link) throws Exception {
141 		log.debug("Trying to establish a link to '" + remoteClass.getSimpleName() + "' API server from daemon at '" + headInfo.getLocation() + "'...");
142 		return (RT) lookupTunneledServer(remoteClass, link, headInfo.getRegistryPort(), DEFAULT_TIMEOUT);
143 	}
144 	
145 	public <RT extends Remote> RT lookupTunneledServer(Class<RT> remoteClass, URILink link, int registryPort, int timeout) throws Exception {
146 		String uriStr = new HeadResourceInfo("127.0.0.1", registryPort).getLocation() + getRegistryId(remoteClass);
147 		RT serverAPI = new RemoteResolver<RT>(uriStr, new GWESocketFactory(link)).getRemote(timeout);
148 		return PulsingServerAPIProxyCreator.createProxyIfNecessary(serverAPI, beatingActor);
149 	}
