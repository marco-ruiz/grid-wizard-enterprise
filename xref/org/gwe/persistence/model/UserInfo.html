82  	/**
83  	 * Sets <code>daemonEncryptor</code> equals to the one correlated to the password hash of the newly created user
84  	 * 
85  	 * @param applyingUser
86  	 * @param daemonInfo
87  	 */
88  	public UserInfo(UserInfo applyingUser, HeadResourceInfo daemonInfo) {
89  	    this(applyingUser.name, applyingUser.passwordHash);
90  	    
91  	    // FIRST USER EVER CREATED WILL BE DAEMON's ADMIN.
92      	// Shall execute only once in the whole life of a GWE installation when creating the admin user for the daemon. 
93     		if (daemonInfo.getAdmin() == null) {
94     			daemonInfo.setAdmin(this);
95     			daemonEncryptor = createDaemonEncryptor(this, daemonInfo);
96     		}
97     		
98     		// Encrypt password hash
99          try {
100        		setEncryptedPasswordHash(daemonEncryptor.encrypt(passwordHash));
101         } catch (Exception e) {
102             // TODO Handle this better than just ignoring it
103         	log.fatal("Problems securing password for user " + this.name, e);
104         }
105     }

129 	 * Sets <code>daemonEncryptor</code> when the "applying" user happens to hold a password hash that which correlates to an 
130 	 * encryptor able to decrypt the daemon's admin encrypted hash password  
131 	 * 
132 	 * @param applyingUser
133 	 * @param daemonInfo
134 	 * @return
135 	 */
136 	public boolean isDaemonUserAdmin(UserInfo applyingUser, HeadResourceInfo daemonInfo) {
137 		Encryptor enc = createDaemonEncryptor(applyingUser, daemonInfo);
138 		boolean result = matchesPassword(applyingUser, enc);
139 		if (result) daemonEncryptor = enc;
140 		return result;
141 	}

172 	private Encryptor getEncryptor() {
173 	    return new Encryptor(getPasswordHash(UserInfo.daemonEncryptor));
174     }

176 	private byte[] getPasswordHash(Encryptor enc) {
177 		String result = getEncryptedPasswordHash();
178 		if (result == null || result.length() == 0 || enc == null) return passwordHash;
179 		
180 		try {
181 			result = enc.decrypt(result.getBytes());
182 		} catch (Exception e) {
183 	        // TODO Handle this better than just ignoring it
184         	log.fatal("Problems retrieving password for user " + this.name, e);
185 		}
186 		return result.getBytes();
187 	}
