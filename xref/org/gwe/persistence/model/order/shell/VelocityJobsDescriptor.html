17  package org.gwe.persistence.model.order.shell;
18  
19  import java.io.Serializable;
20  
21  import org.gwe.GWEAppContext;
22  import org.gwe.p2el.P2ELExpressionContext;
23  import org.gwe.p2el.P2ELFunctionInvocationPhase;
24  import org.gwe.p2el.P2ELStatement;
25  import org.gwe.p2el.P2ELSyntaxException;
26  import org.gwe.persistence.model.UserInfo;
27  import org.gwe.utils.security.KeyStore;
28  
29  /**
30   * 
31   * @author Marco Ruiz
32   * @since Sep 20, 2007
33   */
34  public class VelocityJobsDescriptor<DAEMON_REQUEST_PARAM_TYPE extends Serializable> extends P2ELCommandsJobDescriptor<DAEMON_REQUEST_PARAM_TYPE> {
35  	
36  	protected String stmt;
37  
38  	public VelocityJobsDescriptor() {}
39  
40  	public VelocityJobsDescriptor(P2ELStatement stmtObj) throws P2ELSyntaxException {
41  		this.stmt = stmtObj.runPrecompilationPhase(P2ELFunctionInvocationPhase.INTERPRET).getStatement();
42  	}
43  
44  	protected String getCommandLines(UserInfo user) {
45  		try {
46  			return createPrecompiledStatement(user.getKeyStore()).getEvaluatedVTLScript();
47  		} catch (P2ELSyntaxException e) {
48  			// Impossible. This statement was pre-checked at constructor time. 
49  		}
50  		return "";
51      }
52  
53  	private P2ELStatement createPrecompiledStatement(KeyStore keys) throws P2ELSyntaxException {
54  	    P2ELExpressionContext ctx = new P2ELExpressionContext(null, keys);
55  	    return GWEAppContext.getP2ELEngine().createStatement(ctx, stmt)
56  	    	.runPrecompilationPhase(P2ELFunctionInvocationPhase.EXPAND)
57  	    	.runPrecompilationPhase(P2ELFunctionInvocationPhase.PREEVALUATE);
58      }
59  	
60  	public String toString() {
61  		return stmt;
62  	}
63  }
