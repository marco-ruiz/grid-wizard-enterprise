41  public class ShellCommandDaemonRequest<DRP_TYPE extends Serializable> extends DaemonRequest<DRP_TYPE> {
42  
43  	private static Log log = LogFactory.getLog(ShellCommandDaemonRequest.class);
44  
45  	protected String localizedCmd;
46  	protected String workspacePath;
47  	protected URILink localLink;
48  
49  	public void setOSCommand(String localizedCmd) {
50  		this.localizedCmd = localizedCmd;
51  	}
52  
53  	public void setWorkspacePath(String workspacePath) {
54  		this.workspacePath = workspacePath;
55  	}
56  
57  	public final Serializable process(String workspace) throws Exception {
58  		setup();
59  		String result = workspace + "result-" + jobId + ".out";
60  		FileOutputStream fos = null;
61  		try {
62  			log.info("Preparing daemon request execution.\n\tLocalized command: '" + localizedCmd + "' \n\tWorkspace: '" + workspacePath + "'");
63  			if (localizedCmd != null && !localizedCmd.equals("")) {
64  				ShellCommand cmd = new ShellCommand(localizedCmd, workspacePath, null);
65  				fos = new FileOutputStream(result, false);
66  				cmd.setOutputStream(fos);
67  				GWEAppContext.getGridNetworkAccess().createHandle(localLink).runCommand(cmd);
68  			}
69  		} catch (FileNotFoundException e) {
70  			log.warn("Result file could not be created", e);
71  			// TODO Throw agent level exception, OS cannot create an output stream
72  		} catch (HandleOperationException e) {
73          	log.warn("IMPOSSIBLE");
74  			// TODO: Throw process level exception
75  		} catch (ConnectorException e) {
76  			// Impossible. LocalConnection does not throw ConnectorException 
77  			// (this exception is just inherited from the abstract definition of this method)
78          	log.fatal("IMPOSSIBLE");
79  		}
80  		if (fos != null)
81  			try {
82  				fos.close();
83  			} catch (IOException e) {
84  				log.warn("Result file could not be closed", e);
85  			}
86  		tearDown();
87  		return result;
88  	}
89  
90  	protected Map<String, String> getEnvironment() { return new HashMap<String, String>(); }
91  
92  	protected void setup() {}
93  
94  	protected void tearDown() {}
95  	
96  	protected String createUnderWorkspace(String fileName, String contents) {
97  		String fullFileName = workspacePath + fileName;
98          try {
99      		FileOutputStream fos = new FileOutputStream(fullFileName, false);
100 			fos.write(contents.getBytes());
101 			return fullFileName;
102         } catch (IOException e) {
103 			// TODO: throw a real exception
104         	log.warn("Couldn't create file '" + fullFileName + "'", e);
105 			throw new RuntimeException("Exception thrown when agent tried to create a file for a job. Please contact developers", e);
106         }
107 	}
108 
109 	public void extractLocalLink(String host) {
110 		try {
111 			URILink headLink = keys.generateURILink(new URI("ssh://" + host));
112 	        localLink = new URILink(headLink.getAccountInfo(), new URI("ssh://localhost"));
113         } catch (URISyntaxException e) {
114         	// Impossible
115         	log.fatal("IMPOSSIBLE");
116         }
117     }
