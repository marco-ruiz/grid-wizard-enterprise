23  enum OSType   { APPLE, LINUX, SUN, UNKNOWN, WINDOWS; }
24  enum ARCHType { PPC, PPC_64, SPARC, UNKNOWN, X86, X86_64; }

26  public enum PlatformType {
27  	WIN32(OSType.WINDOWS, ARCHType.X86),
28  	WIN64(OSType.WINDOWS, ARCHType.X86_64),
29  	
30  	MAC_X86(OSType.APPLE, ARCHType.X86),
31  	MAC_PPC(OSType.APPLE, ARCHType.PPC), 
32  	MAC_PPC_64(OSType.APPLE, ARCHType.PPC_64), 
33  	
34  	LINUX_X86(OSType.LINUX, ARCHType.X86), 
35  	LINUX_X86_64(OSType.LINUX, ARCHType.X86_64), 
36  	LINUX_PPC(OSType.LINUX, ARCHType.PPC),
37  	LINUX_PPC_64(OSType.LINUX, ARCHType.PPC_64),
38  	LINUX_SPARC(OSType.LINUX, ARCHType.SPARC), 
39  	
40  	SOLARIS_INTEL(OSType.SUN, ARCHType.X86_64),  
41  	SOLARIS_SPARC(OSType.SUN, ARCHType.SPARC),

44  	AIX(OSType.WINDOWS, ARCHType.X86_64),

46  	HP_UX(OSType.WINDOWS, ARCHType.X86_64),

50  	UNKNOWN(null, null);

52  	private OSType osType;
53  	private ARCHType archType;
54  	
55  	PlatformType(OSType osType, ARCHType archType) {
56  		this.osType = osType;
57  		this.archType = archType;
58  	}
59  	
60  	public static PlatformType detectRunningPlatform() {
61  		OSType runningOS = detectOS();
62  		ARCHType runningArch = detectArch();
63  		
64  		for (PlatformType platform : PlatformType.values()) {
65  	        if (platform.osType != null && platform.archType != null && 
66  	        	platform.osType.equals(runningOS) && platform.archType.equals(runningArch)) 
67  	        	return platform;
68          }
69  		return UNKNOWN;
70  	}
71  	
72  	public static PlatformType getByName(String name) {
73  		for (PlatformType platform : PlatformType.values()) {
74  	        if (name.equals(platform.name())) 
75  	        	return platform;
76          }
77  		return UNKNOWN;
78  	}
79  	
80  	private static OSType detectOS() {
81  		String osName = System.getProperty("os.name").toLowerCase();
82  		assert osName != null;
83  		if (osName.startsWith("mac os x")) 	return OSType.APPLE;
84  		if (osName.startsWith("windows")) 	return OSType.WINDOWS;
85  		if (osName.startsWith("linux")) 	return OSType.LINUX;
86  		if (osName.startsWith("sun")) 		return OSType.SUN;
87  		return OSType.UNKNOWN;
88  	}
89  
90  	private static ARCHType detectArch() {
91  		String osArch = System.getProperty("os.arch").toLowerCase();
92  		assert osArch != null;
93  		if (osArch.equals("x86") || 
94  			osArch.equals("i386")) 			return ARCHType.X86;
95  		if (osArch.startsWith("amd64") || 
96  			osArch.startsWith("x86_64")) 	return ARCHType.X86_64;
97  		if (osArch.equals("ppc")) 			return ARCHType.PPC;
98  		if (osArch.startsWith("ppc")) 		return ARCHType.PPC_64;
99  		if (osArch.startsWith("sparc")) 	return ARCHType.SPARC;
100 		return ARCHType.UNKNOWN;
101 	}

104 		return null;
105 	}
