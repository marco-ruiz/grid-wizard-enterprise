59  	public UserInfo getPersistedUser(UserInfo user, boolean createIfNonExistent) {
60  		UserInfo persistedUser = userDAO.get(user.getName());
61  		if (persistedUser == null) {
62  			if  (!createIfNonExistent) throw new NonExistentUserException();
63  			userDAO.save(user);
64  			persistedUser = user;
65  		}
66  		if (persistedUser.getPassword().equals("")) persistedUser.setPassword(user.getPassword());
67  		if (!persistedUser.getPassword().equals(user.getPassword())) throw new PasswordMismatchException(); 
68  		return persistedUser;
69  	}

163 	public OrderInfo persistOrder(UserInfo user, String gridId, OrderInfo order) {
164 		// Preparation and error checking
165 		user = getPersistedUser(user);
166 		if (order == null) throw new OrderNotSpecifiedException();
167 		if (gridId == null || gridId.equals("")) throw new GridNotSpecifiedException();
168 		GridInfo grid = user.getDefinedGrid(gridId);
169 		if (grid == null) throw new GridNotDefinedByUserException();
170 
171 		// Actual queuing
172 		user.associateOrder(order, grid);
173 		orderDAO.save(order);
174 		return order;
175 	}

190 		List<OrderInfo> newOrdersCopy = newOrders.takeAll();
191 		HeadResourceInfo daemonInfo = headResourceDAO.getDaemonInfo();
192 		for (OrderInfo order : newOrdersCopy) {
193 			List<JobInfo> generateJobs = order.generateJobs(daemonInfo);
194 			jobDAO.saveAll(generateJobs);
195 		}
196 		orderDAO.saveOrUpdateAll(newOrdersCopy);
