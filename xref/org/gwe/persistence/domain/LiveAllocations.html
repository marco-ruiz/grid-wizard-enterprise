17  package org.gwe.persistence.domain;
18  
19  import java.util.HashMap;
20  import java.util.List;
21  import java.util.Map;
22  
23  import org.gwe.persistence.model.AllocationInfo;
24  import org.gwe.persistence.model.JobInfo;
25  import org.gwe.utils.concurrent.BlockingMap;
26  
27  /**
28   * 
29   * @author Marco Ruiz
30   * @since Nov 15, 2007
31   */
32  public class LiveAllocations {
33  
34  	private Map<Integer, AllocationInfo> busyAllocations = new HashMap<Integer, AllocationInfo>();
35  	private BlockingMap<Integer, AllocationInfo> readyAllocations = new BlockingMap<Integer, AllocationInfo>();
36  	
37  	public void saveLiveAllocations(List<AllocationInfo> allocList) {
38  		synchronized (readyAllocations) {
39  			for (AllocationInfo alloc : allocList) busyAllocations.put(alloc.getId(), alloc);
40  		}
41  	}
42  
43  	public void saveLiveAllocation(AllocationInfo alloc) {
44  		synchronized (readyAllocations) {
45  			busyAllocations.put(alloc.getId(), alloc);
46  		}
47  	}
48  
49  	public AllocationInfo getBusyAllocation(int allocId) throws AllocationNotFoundException {
50  		synchronized (readyAllocations) {
51  			AllocationInfo alloc = busyAllocations.get(allocId);
52  			if (alloc == null) throw new AllocationNotFoundException(allocId); 
53  			return alloc;
54  		}
55  	}
56  	
57  	public AllocationInfo removeAllocation(int allocId) {
58  		synchronized (readyAllocations) {
59  			AllocationInfo busyAlloc = busyAllocations.remove(allocId);
60  			return (busyAlloc != null) ? busyAlloc : readyAllocations.remove(allocId); 
61  		}
62  	}
63  
64  	public int getLiveAllocationsCount() {
65  		synchronized (readyAllocations) {
66  			return busyAllocations.size() + readyAllocations.size();
67  		}
68  	}
69  
70  	public void flagAsReady(AllocationInfo alloc) {
71  		synchronized (readyAllocations) {
72  			Integer allocId = alloc.getId();
73  			readyAllocations.put(allocId, busyAllocations.remove(allocId));
74  			alloc.startIdleCountdown();
75  		}
76  	}
77  
78  	public void pairNextReadyAllocationWithJob(JobInfo job) {
79  		synchronized (readyAllocations) {
80  			AllocationInfo alloc;
81  			while (true) {
82  				alloc = readyAllocations.takeOne();
83  				assert alloc != null;
84  				if (alloc.setProcessingJob(job)) break;
85  			}
86  			busyAllocations.put(alloc.getId(), alloc);
87  		}
88  	}
89  }
