49  public class ShellApp extends ShellUserInfoCapturingApp {
50  
51  	private static Log log = LogFactory.getLog(ShellApp.class);

53  	//=================================================
54  	// INSTANCE PROPERTIES TO BE DYNAMICALLY POPULATED
55  	//=================================================
56  	public ShellApp(int index, ArgsList argsList) throws CredentialNotFoundException, ClusterNotSpecifiedException, IOException {
57  		super(System.out, index, argsList);
58  	}
59  	
60  	protected String processCommand(String[] args) {
61  		try {
62  	        return processCommandInternal(args);
63          } catch (ServerAPIConnectionException e) {
64          	return e.getMessage();
65          } catch (RemoteException e) {
66          	return "Unexpected error encountered on the daemon: " + e.getMessage();
67  	    } catch (Exception e) {
68  	    	return e.toString();
69  	    }
70      }
71  	
72  	protected String processCommandInternal(String[] args) throws ServerAPIConnectionException, P2ELSyntaxException, PasswordMismatchException, OrderNotSpecifiedException, GridNotSpecifiedException, GridNotFoundException, RemoteException, GWEDomainException {
73  		String cmdName = args[0];
74  		String[] actualArgs = StringUtils.removeArgs(args, 0, 1);
75  		String output = "Command '" + cmdName + "' not supported";
76  		
77  		ServerAPI4ClientSession session = getSession();
78  		if ("queue-order".equals(cmdName)) {
79  			P2ELStatement stmt = appConfig.createStatement(StringUtils.getArrayAsStr(actualArgs));
80  			OrderInfo order = new OrderInfo(new VelocityJobsDescriptor<String>(stmt));
81  			output = "Order queued with id " + session.queueOrder(order);
82  		}
83  		
84  		if (ClientOrderBasedQuery.getQuery(cmdName) != null)
85  			output = session.doOrderBasedQuery(ClientOrderBasedQuery.getQuery(cmdName), getIdArg(0, actualArgs), getIdArg(1, actualArgs));
86  		
87  		if (ClientOrderBasedOperation.getOperation(cmdName) != null)
88  			output = session.doOrderBasedOperation(ClientOrderBasedOperation.getOperation(cmdName), getIdArg(0, actualArgs));
89  		
90  		return output;
91      }
92  	
93      protected List<String> getCommandList() {
94  		List<String> result = new ArrayList<String>();
95  		result.add("queue-order");
96  		result.addAll(ClientOrderBasedOperation.getOperations());
97  		result.addAll(ClientOrderBasedQuery.getQueries());
98  		return result;
99  	}
100     
101 	private int getIdArg(int idx, String[] args) {
102 		try {
103 			return (idx > args.length - 1) ? -1 : Integer.parseInt(args[idx]);
104 		} catch(NumberFormatException e) {
105 			throw new RuntimeException("Argument '" + args[idx] + "' is not an integer");
106 		}
107 	}
108 }
