61  	public XMLClientConfigReader(PasskeysReader passKeysReader, String confArg) throws CredentialNotFoundException, IOException {
62  		// Format: -conf=[OPTIONAL_CLUSTER_NAME]@[OPTIONAL_CLUSTERS_FILENAME]:[OPTIONAL_KEYSTORE_FILENAME]
63  		OptionParser option = new OptionParser(CONF_ARG_PREFIX, confArg, new char[]{'@', ':'});
64  		daemonName = option.getEle(0, daemonName);
65  		gridFN     = option.getEle(1, gridFN);
66  		keysFN     = option.getEle(2, keysFN);
67  /*
68  		KeyStore keys = readXMLObject(keysFN, KeyStore.class, "keystore", "accessControls", AccessControl.class, "accessControl");
69  		GridInfo grid = readXMLObject(gridFN, GridInfo.class, "grid", "headResources", HeadResourceInfo.class, "cluster");
70  */
71  		ATransf<GridInfo>         at4Grid    = new ATransf<GridInfo>        (GridInfo.class,         "grid", "headResources");
72  		ATransf<HeadResourceInfo> at4HeadRes = new ATransf<HeadResourceInfo>(HeadResourceInfo.class, "cluster", "vars");
73  		ATransf<VarInfo>          at4Var     = new ATransf<VarInfo>         (VarInfo.class,          "p2elVar");
74  		
75  		ATransf<KeyStore>         at4Keys    = new ATransf<KeyStore>        (KeyStore.class,         "keystore", "accessControls");
76  		ATransf<AccessControl>    at4ACs     = new ATransf<AccessControl>   (AccessControl.class,    "accessControl");
77  		ATransf<AccountInfo>      at4Account = new ATransf<AccountInfo>     (AccountInfo.class,      "account");
78  		ATransf<Realm>            at4Realms  = new ATransf<Realm>           (Realm.class,            "realm");
79  
80  //		at4Var.setAttributize(false);
81  		at4ACs.setAttributize(false);
82  
83  		GridInfo grid = readXMLObject(gridFN, at4Grid, at4HeadRes, at4Var);
84  		headResource = grid.getHeadResource(daemonName);
85      	if (headResource == null && grid.getHeadResources().size() > 0) headResource = grid.getHeadResources().get(0);
86  
87  		KeyStore keys = readXMLObject(keysFN, at4Keys, at4ACs, at4Account, at4Realms);
88  		keys.init(passKeysReader);
89  		
90  		this.clientConfiguration = new ClientConfig(grid, keys);
91  	}

93  	private <RESULT_TYPE> RESULT_TYPE readXMLObject(String fileName, ATransf... transformations) throws FileNotFoundException {
94          XStream result = new XStream(new DomDriver());
95          for (ATransf transf : transformations) transf.transform(result);
96  //        result.registerConverter(new RealmAccessConverter());
97          
98  		InputStream is = getClass().getClassLoader().getResourceAsStream(fileName);
99  		if (is == null) is = new FileInputStream(new File(fileName));
100         return (RESULT_TYPE)result.fromXML(is);
101 	}

103 	class ATransf<CLASS_TYPE> {
104 		private Class<CLASS_TYPE> aliasClass;
105 		private String aliasName;
106 		private String implicitCollectionName;
107 		private boolean attributize = true;
108 		
109 		public ATransf(Class<CLASS_TYPE> aliasClass, String aliasName) {
110 			this(aliasClass, aliasName, null);
111         }
112 		
113 		public ATransf(Class<CLASS_TYPE> aliasClass, String aliasName, String implicitCollectionName) {
114 	        this.aliasClass = aliasClass;
115 	        this.aliasName = aliasName;
116 	        this.implicitCollectionName = implicitCollectionName;
117         }
118 
119 		public void setAttributize(boolean attributize) {
120         	this.attributize = attributize;
121         }
122 
123 		public void transform(XStream xstream) {
124 	        xstream.alias(aliasName, aliasClass);
125 	        if (implicitCollectionName != null) xstream.addImplicitCollection(aliasClass, implicitCollectionName);
126 	        if (!attributize) return;
127 	        for (Field fld : aliasClass.getDeclaredFields()) {
128 	        	if (!fld.getName().equals(implicitCollectionName)) 
129 	        		xstream.useAttributeFor(fld.getName(), fld.getType());
130 	        }
131 		}
132 	}

140     }
