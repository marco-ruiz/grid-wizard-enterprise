94  	public <T extends JComponent> T addRow(JComponent compLeft, T compRight) {
95          if ((compLeft == null) && (compRight == null)) return null;
96  
97          if ((compLeft != null) && (compRight != null)) {
98              addColumnComp(compLeft, 0, false);
99              addColumnComp(compRight, 1, false);
100         } else {
101             JComponent comp = (compLeft != null) ? compLeft : compRight;
102             addColumnComp(comp, 0, true);
103         }
104         rows++;
105         
106         return compRight;
107     }
108 
109     private void addColumnComp(JComponent comp, int col, boolean merge) {
110         constraints.gridx = col;
111         constraints.gridy = rows;
112         constraints.gridwidth = merge ? GridBagConstraints.REMAINDER : 1;
113         constraints.weightx = (col == 0) ? 0 : 1;
114         gridbag.setConstraints(comp, constraints);
115         add(comp);
116     }
