17  package org.gwe.app.client.web;
18  
19  import java.lang.reflect.Method;
20  
21  /**
22   * @author Marco Ruiz
23   * @since Jan 31, 2009
24   */
25  public class WebUtils {
26  
27  	public static void openURL(String url) {
28  		String osName = System.getProperty("os.name");
29  		try {
30  			if (osName.startsWith("Mac OS")) {
31  				Class fileMgr = Class.forName("com.apple.eio.FileManager");
32  				Method openURL = fileMgr.getDeclaredMethod("openURL", new Class[] { String.class });
33  				openURL.invoke(null, new Object[] { url });
34  			} else if (osName.startsWith("Windows"))
35  				Runtime.getRuntime().exec("rundll32 url.dll,FileProtocolHandler " + url);
36  			else { // assume Unix or Linux
37  				String[] browsers = { "firefox", "opera", "konqueror", "epiphany", "mozilla", "netscape" };
38  				String browser = null;
39  				for (int count = 0; count < browsers.length && browser == null; count++)
40  					if (Runtime.getRuntime().exec(new String[] { "which", browsers[count] }).waitFor() == 0)
41  						browser = browsers[count];
42  				if (browser == null)
43  					throw new Exception("Could not find web browser");
44  				else
45  					Runtime.getRuntime().exec(new String[] { browser, url });
46  			}
47  		} catch (Exception e) {
48  			// Ignore
49  		}
50  	}
51  }
