99  class HtmlJobPropsTable extends HtmlModelPropsTable<JobInfo> {
100 
101 	public HtmlJobPropsTable(JobInfo job) { super(job); }
102 
103 	protected void populate() {
104 	    JobExecutionInfo exec = modelObj.getExecution();
105         addPropertyRow("Job Id", modelObj.getId());
106         addPropertyRow("Order Id", modelObj.getOrderId());
107         addPropertyRow("Job Number", modelObj.getJobNum());
108         addPropertyRow("Executions", trials(modelObj.getFailures() + 1));
109 	    addPropertyRow("Status", (exec == null) ? "WAITING" : exec.getStatus());
110     }
111 	
112 	private StringBuffer trials(int count) {
113 		StringBuffer result = new StringBuffer(count + " [ ");
114 		for (int idx = 1; idx < count + 1; idx++)
115 			result.append(getExecAnchor(idx) + " ");
116 		return result.append("]");
117 	}
118 
119 	private String getExecAnchor(int number) {
120 	    return "<a href=\"#" + number + "\">" + number + "</a>";
121     }
122 }
