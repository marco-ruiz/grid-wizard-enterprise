75  		if (install) enablePasswordConfirmation();

77  		UserInfo user = gatherGWEUser(install, launch);

80  		if (update)  updateDaemon();
81  		if (install) installDaemon();

137 	public boolean installDaemon() {
138 		String homeFolder = ClientAppConfig.extractGWE_HOME();
139 		if (homeFolder == null) {
140 			println("'GWE_HOME' environment variable not set. Unable to locate GWE distribution to install. Exiting.");
141 			return false;
142 		}
143 		String separator = System.getProperty("file.separator");
144 		if (!homeFolder.endsWith(separator)) homeFolder += separator;
145 		println("Deploying bundle containing GWE daemon's distribution (this may take a few minutes)...");
146 		boolean newlyDeployed;
147 		try {
148 			newlyDeployed = daemonAppManager.deployDaemon(OSAppFolder.BUNDLES.getDirRelative(homeFolder));
149 		} catch (HandleOperationException e) {
150 			println("GWE daemon failed to launch. Problems establishing network connection to remote compute node.\n\t" + e.getMessage());
151 			return false;
152 		} catch (ConnectorException e) {
153 			println("GWE daemon failed to launch. Problems executing command to unbundle GWE distribution on remote compute node.\n\t" + e.getMessage());
154 			return false;
155 		} catch (Exception e) {
156 			println("GWE daemon failed to deploy. Problems transfering GWE distribution bundle to remote compute node.\n\t" + e.getMessage());
157 			return false;
158 		}
159 		
160 		if (!newlyDeployed) {
161 			println("GWE daemon was already deployed.");
162 		} else {
163 			println("GWE daemon deployed. Configuring GWE daemon (this may take a few minutes)...");
164 			try {
165 				daemonAppManager.configureDaemon();
166 			} catch (Exception e) {
167 				processSystemException("configure", e);
168 				return false;
169 			}
170 			println("GWE daemon fully installed and ready to be launched.");
171 		}
172 		
173 		return true;
174 	}

176 	public boolean launchDaemon(UserInfo admin, boolean tryToStartFirst) {
177 		ServerAPIAccessor accessor = daemonAppManager.getAppConfig().createAPIAccessor();
178 
179 		if (tryToStartFirst) {
180 			ServerAPI4UserAdmin link = createAdminLink(accessor);
181 			if (link != null) return startDaemon(admin, link);
182 		}
183 		
184 		println("Launching GWE daemon (this may take a few minutes)...");
185 		try {
186 			daemonAppManager.launchDaemon();
187 			println("GWE daemon launched.");
188 		} catch (LaunchFailedException e) {
189 			processSystemException("launch", e);
190 		} catch (Exception e) {
191 			processSystemException("launch", e);
192 			return false;
193 		}
194 		
195 		ServerAPI4UserAdmin link = createAdminLink(accessor);
196 		return (link != null && startDaemon(admin, link));
197 	}
198 	
199 	private ServerAPI4UserAdmin createAdminLink(ServerAPIAccessor accessor) {
200 	    try {
201 	        return accessor.createLink(ServerAPI4UserAdmin.class);
202         } catch (Exception e) {
203 	    	println("Unable to contact GWE daemon OS process. It may not be running.");
204         	return null;
205         }
206     }
207 
208 	private boolean startDaemon(UserInfo admin, ServerAPI4UserAdmin link) {
209 	    println("Trying to start GWE daemon services...");
210 		try {
211 			link.startDaemon(admin);
212 			return true;
213 		} catch (UndeclaredThrowableException e) {
214 			processStartDaemonException(e.getUndeclaredThrowable());
215 		} catch (Exception e) {
216 			processStartDaemonException(e);
217 		}
218 		return false;
219     }
220 	
221 	private void processStartDaemonException(Throwable exc) {
222 		if (exc instanceof PasswordMismatchException || exc instanceof UserIsNotDaemonAdminException) {
223 			println("GWE daemon refused to start. User provided couldn't be authenticated as daemon's admin. User is not GWE daemon's admin or the password is incorrect.");
224 		} else {
225 			println("GWE daemon failed to start. Most likely another instance is already running.");
226 		}
227 	}
228 
229 	private void processSystemException(String action, Exception e) {
230 		String mainMsg = "GWE daemon failed to " + action + ". ";
231 		
232 		if (e instanceof HandleOperationException) {
233 			println(mainMsg + "Problems accessing remote compute node.\n\t" + e.getMessage());
234 		} else if (e instanceof ConnectorException) {
235 			println(mainMsg + "Problems executing launchable command on remote compute node.\n\t" + e.getMessage());
236 		} else if (e instanceof LaunchFailedException) {
237 			println(mainMsg + "Most likely an instance of GWE daemon is already running on the target remote compute node.");
238 		}
239 	}
