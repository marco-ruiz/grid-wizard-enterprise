63  	@Override
64  	protected int submitOrderInternal(String userName, String password, String clusterName, boolean autoInstall, String script, boolean isLDDMMOrder) {
65  		tracker = new RichProgressTracker(txtArea);
66  		UserInfo user = new UserInfo(userName, password);
67  		try {
68  			VelocityJobsDescriptor jd = new VelocityJobsDescriptor(new P2ELStatement(script));
69  			jd.setDaemonRequestClass(ShellCommandDaemonRequest.class);
70  			
71  			// TODO: FIX!!!
72  //			if (isLDDMMOrder) jd = ClientAppConfig.getInstance().getBean("LDDMMTestJobs");
73  
74  			OrderInfo order = new OrderInfo(jd);
75  
76  			HeadResourceInfo daemonInfo = getCluster(clusterName);
77  			if (coordinator == null) connect(user, daemonInfo);
78  
79  			GridInfo grid = getDefinedGridFor(user, daemonInfo);
80  
81  			tracker.trackProgress("Requesting order processing...");
82  			int id = coordinator.queueOrder(user, grid.getId(), order);
83  
84  			tracker.trackProgress("Order accepted by daemon. Tracking id: " + id);
85  			return id;
86  		} catch (Exception e) {
87  			tracker.trackProgress("Problem encountered while preparing to submit the order: " + e.getMessage());
88  		}
89  		
90  		tracker.shutdown();
91  		return 0;
92  	}

139 	private HeadResourceInfo getCluster(String name) {
140 		for (HeadResourceInfo cl : clusters) 
141 			if (name.equals(cl.getName())) return cl;
142 		return null;
143 	}
144 
145 	@Override
146 	protected Object[] defineClusterInternal(String name, String host, int maxCN, long maxHijack, long maxIdle) {
147 		HeadResourceInfo cl = new HeadResourceInfo(host, 1099);
148 		cl.setMaxHijackMillis(maxHijack);
149 		cl.setMaxIdleMillis(maxIdle);
150 		cl.setQueueSize(maxCN);
151 		cl.setName(name);
152 		clusters.add(cl);
153 
154 		Object[] results = new Object[clusters.size()];
155 		for (int idx = 0; idx < results.length; idx++) results[idx] = clusters.get(idx).getName();
156 		return results;
157 	};
