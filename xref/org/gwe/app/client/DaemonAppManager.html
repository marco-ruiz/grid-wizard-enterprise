36  /**
37   * @author Marco Ruiz
38   * @since Dec 11, 2007
39   */
40  public class DaemonAppManager {
41  
42  	private static Log log = LogFactory.getLog(DaemonAppManager.class);
43  
44  	private static final String DAEMON_APP_CONFIGURED_MESSAGE = "DAEMON_CONFIGURED_SUCCESSFULLY";
45      private static final String DAEMON_APP_READY_MESSAGE = "DAEMON_READY_TO_SERVE_REQUESTS";
46  
47  	// Services required 
48  	private GridBundleManager bundleMgr;
49  	private GridOSAppManager  appMgr;
50  	
51  	// Daemon info
52  	private ClientAppConfig appConfig;
53  	
54  	public DaemonAppManager(ClientAppConfig cfg) throws URISyntaxException {
55  		this.appConfig = cfg;
56  		bundleMgr = GWEAppContext.getInstance().getBeanOfClass(GridBundleManager.class);
57  		appMgr    = GWEAppContext.getInstance().getBeanOfClass(GridOSAppManager.class);
58  		
59  		JavaOSAppHandle osApp = appMgr.createHandle(cfg.getNodeAccount().getHomeDir(), DaemonApp.class, true);
60  		appConfig.getClusterSelected().setDaemonOSApp(osApp);
61  //		daemonOSApp = osAppFactory.createHandle(installationPath, OSAppType.DAEMON);
62  //		daemonOSApp = osAppFactory.createHandle(machineLink.getAccountInfo().getHomeDir(), OSAppType.DAEMON);
63  	}
64  
65  	public ClientAppConfig getAppConfig() {
66  		return appConfig;
67  	}
68  
69  	public void installDaemon() throws Exception {
70  		installDaemon(null);
71  	}
72  	
73  	public void installDaemon(String sourceFolder) throws Exception {
74  		if (deployDaemon(sourceFolder)) configureDaemon();
75  	}
76  
77  	public boolean deployDaemon(String sourceFolder) throws Exception {
78  		URILink destLink = appConfig.getNodeLink();
79  		if (sourceFolder == null || "".equals(sourceFolder)) sourceFolder = OSAppFolder.BUNDLES.getUserDirRelative();
80  		
81  		// Deploy daemon
82  		JavaOSAppHandle daemonOSApp = appConfig.getClusterSelected().getDaemonOSApp();
83  		BundleHandle bundle = bundleMgr.createHandle(BundleType.ZIP, sourceFolder, daemonOSApp.getVersionedName("-daemon"));
84  		
85  		return bundle.deploy(destLink, daemonOSApp.getFolderFullPath(OSAppFolder.ROOT), daemonOSApp.getFolderFullPath(OSAppFolder.HOME), appConfig.getKeys());
86  	}
87  
88  	public void configureDaemon() throws HandleOperationException, ConnectorException, LaunchFailedException {
89  		URILink destLink = appConfig.getNodeLink();
90  		// Deploy serialized link for agents and configure daemon
91  		String filePath = appConfig.getClusterSelected().getLinkSerializedFilePath();
92  		String destFile = destLink.createSFTPURI(filePath);
93  		GWEAppContext.getGridFileSystem().createHandle(destFile, appConfig.getKeys()).storeObject(destLink);
94  		executeDaemonRemotely(true); // Run daemon app remotely in configuration mode (create DB and initial records)
95  	}
96  	
97  	public void launchDaemon() throws HandleOperationException, ConnectorException, LaunchFailedException {
98  		executeDaemonRemotely(false);
99  	}
100 	
101 	private void executeDaemonRemotely(boolean configMode) throws HandleOperationException, ConnectorException, LaunchFailedException {
102 		JavaOSAppHandle daemonOSApp = appConfig.getClusterSelected().getDaemonOSApp();
103 		String exitToken = configMode ? DAEMON_APP_CONFIGURED_MESSAGE : DAEMON_APP_READY_MESSAGE;
104 		String[] args = new DaemonApp().createAppArgs(configMode, appConfig.getClusterSelected(), exitToken);
105 		daemonOSApp.setArgs(args);
106 		daemonOSApp.launch(appConfig.getNodeLink(), 120000, exitToken);
107 	}
108 }
