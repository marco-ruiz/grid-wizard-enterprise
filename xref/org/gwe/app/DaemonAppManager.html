53  		if (deployDaemon(sourceFolder)) configureDaemon();
54  	}

56  	public boolean deployDaemon(String sourceFolder) throws Exception {
57  		URILink machineLink = daemonLogin.getNodeLink();
58  		if (sourceFolder == null || "".equals(sourceFolder)) sourceFolder = OSAppFolder.BUNDLES.getUserDirRelative();
59  		
60  		// Deploy daemon
61  		JavaOSApp daemonOSApp = daemonLogin.getClusterSelected().getDaemonOSApp();
62  		Bundle appBundle = new Bundle(BundleType.ZIP, sourceFolder, daemonOSApp, "-daemon");
63  		return packageMgr.deploy(appBundle, machineLink, daemonOSApp, daemonLogin.getKeys());
64  	}
65  
66  	public void configureDaemon() throws HandleOperationException, ConnectorException, LaunchFailedException {
67  		URILink machineLink = daemonLogin.getNodeLink();
68  		// Deploy serialized link for agents and configure daemon
69  		packageMgr.deployObjectSerialized(machineLink, machineLink, daemonLogin.getClusterSelected().getLinkSerializedFilePath(), daemonLogin.getKeys());
70  		executeDaemonRemotely(true); // Run daemon app remotely in configuration mode (create DB and initial records)
71  	}
72  	
73  	public void launchDaemon() throws HandleOperationException, ConnectorException, LaunchFailedException {
74  		executeDaemonRemotely(false);
75  	}
76  	
77  	private void executeDaemonRemotely(boolean configMode) throws HandleOperationException, ConnectorException, LaunchFailedException {
78  		JavaOSApp daemonOSApp = daemonLogin.getClusterSelected().getDaemonOSApp();
79  		String exitToken = configMode ? DAEMON_APP_CONFIGURED_MESSAGE : DAEMON_APP_READY_MESSAGE;
80  		String[] args = new DaemonApp().createAppArgs(configMode, daemonLogin.getClusterSelected(), exitToken);
81  		daemonOSApp.setArgs(args);
82  		packageMgr.launch(daemonLogin.getNodeLink(), daemonOSApp, 120000, exitToken);
83  	}
