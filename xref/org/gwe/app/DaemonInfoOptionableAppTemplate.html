28  	public HeadResourceInfo createDaemonInfoUsingArgs(OSAppFactory appFactory) {
29  		HeadResourceInfo daemonInfo = new HeadResourceInfo(getArg(OPT_HOST), getArgAsInt(OPT_PORT));
30  		daemonInfo.setName(getArg(OPT_NAME));
31  		
32  //		Long age = getMinArgAsMillis(OPT_AGE);
33  		Long age = getArgAsLong(OPT_AGE);
34  		if (age != null) daemonInfo.setMaxHijackMillis(age);
35  		
36  //		Long idle = getMinArgAsMillis(OPT_IDLE);
37  		Long idle = getArgAsLong(OPT_IDLE);
38  		if (idle != null) daemonInfo.setMaxIdleMillis(idle);
39  		
40  		Integer size = getArgAsInt(OPT_SIZE);
41  		if (size != null) daemonInfo.setQueueSize(size); 
42  
43  		daemonInfo.setDaemonOSApp(createDaemonOSAppUsingArgs(appFactory));
44  		return daemonInfo;
45  	}
