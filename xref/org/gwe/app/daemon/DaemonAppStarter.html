17  package org.gwe.app.daemon;
18  
19  import java.util.HashSet;
20  import java.util.Map;
21  import java.util.Set;
22  
23  import org.gwe.api.exceptions.ServerAPIBindingException;
24  import org.gwe.api.impl.BaseServerAPIImpl;
25  import org.gwe.persistence.model.DaemonConfigDesc;
26  import org.gwe.persistence.model.HeadResourceInfo;
27  
28  /**
29   * @author Marco Ruiz
30   * @since Nov 3, 2008
31   */
32  public class DaemonAppStarter {
33  	
34  	private static final Set hackToPreventGarbageCollection = new HashSet();
35  	
36  	private ThreadGroup servicesGroup = new ThreadGroup("Background Services");
37  	
38  	private void startBackgroundService(Runnable serv, String name) {
39  		Thread threadObj = new Thread(servicesGroup, serv, "BG Service: " + name);
40  		threadObj.setDaemon(false);
41  		threadObj.start();
42  	}
43  
44  	private DaemonConfigDesc configuration;
45  	private Set<BaseServerAPIImpl> serverAPIs;
46  	private Map<Runnable, String> daemonThreadServices;
47  
48  	public void setConfiguration(DaemonConfigDesc configuration) {
49      	this.configuration = configuration;
50      }
51  
52  	public void setServerAPIs(Set<BaseServerAPIImpl> serverAPIs) {
53  		this.serverAPIs = serverAPIs;
54  	}
55  
56  	public void setDaemonThreadServices(Map<Runnable, String> daemonThreadServices) {
57  		this.daemonThreadServices = daemonThreadServices;
58  	}
59  
60  	public void registerRMIServerObjects() throws ServerAPIBindingException {
61  		HeadResourceInfo daemonInfo = configuration.getHeadResource();
62  		
63  		// Bind server objects to RMI registry
64  		for (BaseServerAPIImpl<?, ?> api : serverAPIs)
65  			hackToPreventGarbageCollection.add(api.bind(daemonInfo));
66      }
67  
68  	public void startServices() {
69  	    // Start thread services
70  		for (Runnable serv : daemonThreadServices.keySet())
71  			startBackgroundService(serv, daemonThreadServices.get(serv));
72      }
73  }
