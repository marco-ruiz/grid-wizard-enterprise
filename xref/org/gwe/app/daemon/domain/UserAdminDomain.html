69      public void verifyIsDaemonUserAdmin(UserInfo applyingUser) throws PasswordMismatchException, UserNotSpecifiedException, UserIsNotDaemonAdminException {
70      	HeadResourceInfo daemonInfo = headResourceDAO.getDaemonInfo();
71      	UserInfo admin = daemonInfo.getAdmin();
72  		if (admin == null) {
73      		getOrCreateUser(applyingUser);
74      		return;
75      	}
76  		if (!admin.getId().equals(applyingUser.getId())) throw new UserIsNotDaemonAdminException(applyingUser);
77  		if (!admin.isDaemonUserAdmin(applyingUser, daemonInfo)) throw new PasswordMismatchException(applyingUser.getId());
78      }

81  	// Server API Services
82  	//=====================
83  	public void startDaemon(UserInfo admin) throws ServerAPIsBindingException {
84  		synchronized (started) {
85  			if (started) return;
86  			try {
87  				for (BaseServerAPIImpl<?, ?> api : serverAPIs) hackToPreventGarbageCollection.add(bind(api));
88  			} catch (Exception e) {
89  				String msg = "Could not bind daemon APIs to registry";
90  				log.fatal(msg, e);
91  				throw new ServerAPIsBindingException(msg);
92  			}
93  			
94  			for (Runnable serv : daemonThreadServices.keySet())
95  				startBackgroundService(serv, daemonThreadServices.get(serv));
96  			started = true;
97  		}
98      }

104 	public HeadResourceInfo getPersistedDaemon(UserInfo user, String daemonLocation) throws DaemonNotFoundException, UserIsNotDaemonAdminException {
105 		HeadResourceInfo persistedDaemon = headResourceDAO.get(daemonLocation);
106 		if (persistedDaemon == null)
107 			throw new DaemonNotFoundException(daemonLocation);
108 		if (!user.equals(persistedDaemon.getAdmin()))
109 			throw new UserIsNotDaemonAdminException(user);
110 		return persistedDaemon;
111 	}
