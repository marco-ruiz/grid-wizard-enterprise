24   * @author Marco Ruiz
25   * @since Feb 13, 2008

27  public class SchedulingCalculator {
28  
29  	private List<OrderStats> queueStats;
30  	private int totalAmount;
31  	private long now = System.currentTimeMillis();
32  	
33  	public SchedulingCalculator(List<OrderStats> queueStats, int totalAmount) {
34  		this.queueStats = queueStats;
35  		this.totalAmount = totalAmount;
36  	}
37  	
38  	public Map<String, Integer> computeSchedulingStrategy() {
39  		for (int count = 0; count < totalAmount; count++) {
40  			pickSchedulingCandidate().increaseSchedulingAmount();
41  		}
42  		
43  		Map<String, Integer> result = new HashMap<String, Integer>();
44  		for (OrderStats stats : queueStats) {
45  			if (stats.getSchedulingAmount() == 0) continue;
46  			result.put(stats.getUserId(), stats.getSchedulingAmount());
47  		}
48  		
49  		return result;
50  	}
51  
52  	private OrderStats pickSchedulingCandidate() {
53  		int totalWaitFeeUnits = 0;
54  		long totalAvgWaitTime = 0;
55  		for (OrderStats stats : queueStats) {
56  			stats.setNow(now);
57  			totalWaitFeeUnits += stats.getWaitFeeLevel();
58  			totalAvgWaitTime += stats.getWaitTime();
59  		}
60  		
61  		float waitPerWeightUnit = totalAvgWaitTime / totalWaitFeeUnits;
62  
63  		OrderStats candidate = null;
64  		float candidateDebt = 0;
65  		for (OrderStats stats : queueStats) {
66  			if (candidate != null) continue;
67  
68  			float debtTime = stats.getDebtTime(waitPerWeightUnit);
69  			if (debtTime < candidateDebt) {
70  				candidate = stats;
71  				candidateDebt = debtTime;
72  			}
73  		}
74  		return candidate;
75  	}
76  }

78  class OrderStats {
79  	private String userId;
80  	private int waitFeeLevel; // Inverse priority, the more cost an order carries, the more expensive their jobs should be considered
81  	private long avgPendingJobCreationTime;
82  	private int pendingJobs;
83  	private int schedulingAmount = 0;
84  	private long now;
85  
86  	public OrderStats(String userId, int waitFeeLevel, long avgPendingJobCreationTime, int pendingJobs) {
87  		this.userId = userId;
88  		this.waitFeeLevel = waitFeeLevel;
89  		this.avgPendingJobCreationTime = avgPendingJobCreationTime;
90  		this.pendingJobs = pendingJobs;
91  	}
92  	
93  	public void setNow(long now) {
94  		this.now = now;
95  	}
96  
97  	public String getUserId() {
98  		return userId;
99  	}
100 
101 	public int getWaitFeeLevel() {
102 		return waitFeeLevel;
103 	}
104 
105 	public int getPendingJobs() {
106 		return pendingJobs;
107 	}
108 
109 	public int getSchedulingAmount() {
110 		return schedulingAmount;
111 	}
112 
113 	public void increaseSchedulingAmount() {
114 		schedulingAmount++;
115 	}
116 	
117 	public long getWaitTime() {
118 		return now - avgPendingJobCreationTime * (pendingJobs - schedulingAmount) / pendingJobs;
119 	}
120 	
121 	public float getDebtTime(float waitPerWeightUnit) {
122 		return getWaitTime() - waitFeeLevel * waitPerWeightUnit;
123 	}
124 }
