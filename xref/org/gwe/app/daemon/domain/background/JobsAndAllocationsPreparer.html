17  package org.gwe.app.daemon.domain.background;
18  
19  import java.util.List;
20  import java.util.TimerTask;
21  import java.util.concurrent.ExecutorService;
22  
23  import org.apache.commons.logging.Log;
24  import org.apache.commons.logging.LogFactory;
25  import org.gwe.app.daemon.domain.AgentDomain;
26  import org.gwe.drivers.resManagers.GridResourceManager;
27  import org.gwe.persistence.model.AllocationInfo;
28  import org.gwe.persistence.model.JobInfo;
29  import org.gwe.utils.concurrent.ThreadPoolUtils;
30  
31  /**
32   * @author Marco Ruiz
33   * @since Nov 19, 2007
34   */
35  public class JobsAndAllocationsPreparer extends AgentDomainBasedService {
36  
37  	private static Log log = LogFactory.getLog(JobsAndAllocationsPreparer.class);
38  	
39  	private ExecutorService allocCreatorsThreadPool = ThreadPoolUtils.createThreadPool("Allocation Schedulers");
40  	private GridResourceManager gridResourceManager = null;
41  
42  	public JobsAndAllocationsPreparer(AgentDomain agentDom, GridResourceManager gridResourceManager) {
43  		super("Jobs & Allocations Preparer", agentDom);
44  		this.gridResourceManager = gridResourceManager;
45  	}
46  
47  	public void execute() throws Exception {
48  		final List<AllocationInfo> allocations = agentDom.prepareMoreAllocations();
49  		// IMPORTANT: If the preparation cycle is too quick the agent may have time to come back and interfere with these JobInfo updates 
50  		List<JobInfo> jobs = agentDom.selectNextJobBatch();
51  		// IMPORTANT: If the selection cycle is too slow the preparers may finished before the jobs get updated into the DB causing sync issues 
52  		agentDom.launchPreparers(jobs);
53  
54  		log.info("Allocating compute resources: " + allocations);
55  		for (AllocationInfo alloc : allocations) allocComputeResourceAsync(alloc);
56  	}
57  
58  	private void allocComputeResourceAsync(final AllocationInfo alloc) {
59  	    allocCreatorsThreadPool.submit(new Runnable() {
60  	    	public void run() {
61      			// For each allocation queue a compute resource allocation requests in local resource manager 
62  	    		try {
63                  	createTooLateTimer(alloc);
64                  	gridResourceManager.getResourceManagerDriver().allocateComputeResource(alloc);
65                  } catch (Exception e) {
66                  	log.warn("Exception thrown while trying to allocate a compute resource.", e);
67                  }
68  	    	}
69  	    });
70      }
71  
72  	private void createTooLateTimer(final AllocationInfo alloc) {
73  		long maxAttachMillis = agentDom.getConfig().getHeadResource().getMaxAttachMillis();
74  		if (maxAttachMillis == -1) return;
75  		
76  		TimerTask releaseTooLateTask = new TimerTask() {
77  			public void run() {
78  				alloc.flagAsTooLate();
79  				agentDom.triggerMoreJobsAndAllocationPreparation();
80  			}
81  		};
82  			
83  		alloc.setTooLateTimerTask(releaseTooLateTask, maxAttachMillis);
84  	}
85  }
