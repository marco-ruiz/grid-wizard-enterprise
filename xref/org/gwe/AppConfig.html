58  	public static void init(String appWorkspace, Class<? extends AppConfig> cfgClazz, String... contextFiles) {
59  		if (instance != null) return;
60  		if (appWorkspace == null || "".equals(appWorkspace)) appWorkspace = "";
61  		System.setProperty(APPLICATION_WORKSPACE_KEY, appWorkspace);
62  		createLoggerAppender(appWorkspace);
63  		for (int idx = 0; idx < contextFiles.length; idx++) 
64  			contextFiles[idx] = ResourcePatternResolver.CLASSPATH_ALL_URL_PREFIX + contextFiles[idx];
65  		try {
66  			instance = cfgClazz.newInstance();
67  		} catch (Exception e) {
68  			log.fatal("Could not create application configuration object (class '" + cfgClazz + "'");
69  			System.exit(0);
70  		}
71  		instance.initContext(contextFiles);
72  	}
73  
74  	protected static void createLoggerAppender(String appWorkspace) {
75  		String appLog = appWorkspace + "app.log";
76  		try {
77  			Logger.getRootLogger().removeAllAppenders();
78  			Logger.getRootLogger().addAppender(new FileAppender(LOG_LAYOUT, appLog));
79  			Logger.getRootLogger().setLevel(Level.INFO);
80  		} catch (IOException e) {
81  			log.warn("Couldn't create file log appender '" + appLog + "'", e);
82  		}
83  	}
