52  	private String readChannel(ChannelExec channel) throws IOException, JSchException {
53  		InputStream in = channel.getInputStream();
54  		channel.connect(30000);
55  		log.trace("Channel successfully connected");
56  		StringBuffer out = new StringBuffer();
57  		byte[] buff = new byte[1024];
58  		while (true) {
59  			while (in.available() > 0) {
60  				int count = in.read(buff);
61  				log.trace("Read " + count + " bytes");
62  				if (0 > count)
63  					break;
64  				out.append(new String(buff, 0, count));
65  			}
66  			if (channel.isClosed()) {
67  				log.info("Exit status: " + channel.getExitStatus());
68  				return out.toString();
69  			}
70  			try {
71  				Thread.yield();
72  			} catch (Exception ex) {
73  			} // From the Exec example in jsch.
74  		}
